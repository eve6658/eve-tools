import struct, time, sys, ctypes, warnings, json, os
from dataclasses import dataclass
from typing import List, Optional, Tuple

import pymem

import numpy as np

warnings.filterwarnings('ignore')

PROCESS    = "AX.exe"
STOCK_CODE = "01347"
BID1       = 138.07
ASK1       = 138.08
PRICE_LO   = 120.0
PRICE_HI   = 180.0
REC_SIZE   = 20
REC_FMT    = "<fIIII"
CACHE_FILE = os.path.join(os.path.expanduser("~"), "ax_cache.json")

@dataclass
class Rec:
    price: float
    vol: int
    cnt: int
    status: int
    seq: int
    offset: int = 0

pm  = pymem.Pymem(PROCESS)
mod = pymem.process.module_from_name(pm.process_handle, PROCESS)
print(f"[+] {PROCESS} PID={pm.process_id}")

kernel32 = ctypes.windll.kernel32

class MBI(ctypes.Structure):
    _fields_ = [
        ("BaseAddress",       ctypes.c_void_p),
        ("AllocationBase",    ctypes.c_void_p),
        ("AllocationProtect", ctypes.c_ulong),
        ("RegionSize",        ctypes.c_size_t),
        ("State",             ctypes.c_ulong),
        ("Protect",           ctypes.c_ulong),
        ("Type",              ctypes.c_ulong),
    ]

def enum_regions():
    regions = []
    mbi = MBI()
    cur = ctypes.c_size_t(0)
    while cur.value < 0x7FFFFFFFFFFF:
        ret = kernel32.VirtualQueryEx(
            pm.process_handle, ctypes.c_void_p(cur.value),
            ctypes.byref(mbi), ctypes.sizeof(mbi))
        if ret == 0:
            break
        base = mbi.BaseAddress
        sz   = mbi.RegionSize
        if base is None or sz == 0:
            cur.value += 0x1000; continue
        if mbi.State == 0x1000:
            prot = mbi.Protect & 0xFF
            if prot in (2, 4, 6, 8, 0x20, 0x40, 0x60, 0x80):
                if sz >= REC_SIZE * 10:
                    regions.append((base, sz))
        nxt = base + sz
        if nxt <= cur.value:
            cur.value += 0x1000; continue
        cur.value = nxt
    return regions

def read_side(base, size, align):
    """读取一侧数据，按价格去重保留 seq 最大的记录"""
    try:
        raw = pm.read_bytes(base, size)
    except:
        return []
    recs = []
    pos = align
    while pos + REC_SIZE <= len(raw):
        try:
            p, v, c, s, q = struct.unpack_from(REC_FMT, raw, pos)
        except:
            break
        if PRICE_LO <= p <= PRICE_HI and 0 < q < 50000:
            recs.append(Rec(round(p, 4), v, c, s, q, base + pos))
        pos += REC_SIZE
    # ===== 按价格去重：每个价格只保留 seq 最大的那条 =====
    by_price = {}
    for r in recs:
        if r.price not in by_price or r.seq > by_price[r.price].seq:
            by_price[r.price] = r
    return list(by_price.values())

def scan_region(base, size):
    """扫描一个区域的所有对齐，返回最佳候选列表"""
    try:
        raw = pm.read_bytes(base, size)
    except:
        return []

    results = []
    for a in range(REC_SIZE):
        recs = []
        pos = a
        while pos + REC_SIZE <= len(raw):
            try:
                p, v, c, s, q = struct.unpack_from(REC_FMT, raw, pos)
            except:
                break
            if PRICE_LO <= p <= PRICE_HI and 0 < q < 50000:
                recs.append(Rec(round(p, 4), v, c, s, q, base + pos))
            pos += REC_SIZE

        if len(recs) < 50:
            continue

        by_seq = sorted(recs, key=lambda r: r.seq)
        ps = [r.price for r in by_seq]
        asc  = sum(1 for i in range(1, len(ps)) if ps[i] >= ps[i-1])
        desc = sum(1 for i in range(1, len(ps)) if ps[i] <= ps[i-1])
        side = "bid" if desc >= asc else "ask"
        mono = max(asc, desc) / (len(ps) - 1) if len(ps) > 1 else 0

        density = len(recs) / size
        seqs = sorted(set(r.seq for r in recs))
        seq_cont = (sum(1 for i in range(1, len(seqs))
                       if seqs[i] - seqs[i-1] <= 2) / (len(seqs) - 1)
                   if len(seqs) >= 2 else 0)

        score = (min(len(recs) / 150, 1.0) * 30 +
                 mono * 25 + seq_cont * 25 +
                 min(density / 0.01, 1.0) * 20)

        results.append((score, base, size, a, side, len(recs), mono))

    return results

def fast_discover():
    t0 = time.perf_counter()

    print(f"\n[*] 枚举区域...")
    regions = enum_regions()
    print(f"[+] {len(regions)} 个区域")

    small = [(b, s) for b, s in regions if 1000 <= s <= 1024 * 1024]
    print(f"    候选区域 (1KB~1MB): {len(small)} 个")

    tol = max(BID1, ASK1) * 0.05
    print(f"\n[*] 搜索 ±{tol:.2f}...")

    candidates = []
    scanned = 0
    for base, size in small:
        try:
            raw = pm.read_bytes(base, size)
        except:
            continue

        scanned += 1
        arr = np.frombuffer(raw, dtype='<f4')
        mask = (np.isfinite(arr) &
                ((np.abs(arr - BID1) < tol) | (np.abs(arr - ASK1) < tol)))
        if not np.any(mask):
            continue

        results = scan_region(base, size)
        candidates.extend(results)

    dt = (time.perf_counter() - t0) * 1000
    candidates.sort(key=lambda x: -x[0])

    print(f"[+] 扫描 {scanned} 个区域, {len(candidates)} 个候选 ({dt:.0f}ms)\n")

    for i, (sc, base, sz, a, side, n, mono) in enumerate(candidates[:15]):
        print(f"  [{i:2d}] {sc:5.1f} {side:3s} 0x{base:X} a={a:2d} "
              f"{n:4d}条 sz={sz//1024}KB mono={mono:.2f}")

    bid_info = None
    ask_info = None
    for sc, base, sz, a, side, n, mono in candidates:
        if side == "bid" and bid_info is None:
            bid_info = (base, sz, a)
        elif side == "ask" and ask_info is None:
            ask_info = (base, sz, a)
        if bid_info and ask_info:
            break

    return bid_info, ask_info, dt

def save_cache(bid_info, ask_info):
    data = {"bid": list(bid_info), "ask": list(ask_info), "time": time.time()}
    with open(CACHE_FILE, 'w') as f:
        json.dump(data, f)
    print(f"[+] 缓存已保存")

def load_cache():
    if not os.path.exists(CACHE_FILE):
        return None, None
    try:
        with open(CACHE_FILE) as f:
            data = json.load(f)
        return tuple(data["bid"]), tuple(data["ask"])
    except:
        return None, None

def validate_cache(bid_info, ask_info):
    bids = read_side(*bid_info)
    asks = read_side(*ask_info)
    if len(bids) < 50 or len(asks) < 50:
        return False
    bids.sort(key=lambda r: -r.price)
    asks.sort(key=lambda r:  r.price)
    sp = asks[0].price - bids[0].price
    return 0 < sp < 1.0

def print_book(bids, asks, dt_ms=0):
    if not bids or not asks:
        print(f"[!] 数据不完整")
        return
    bids.sort(key=lambda r: -r.price)
    asks.sort(key=lambda r:  r.price)
    sp = asks[0].price - bids[0].price
    md = (asks[0].price + bids[0].price) / 2

    print(f"\n{'='*60}")
    print(f"  {STOCK_CODE}  买一={bids[0].price:.2f} 卖一={asks[0].price:.2f} "
          f"差={sp:.4f}({sp/md*100:.3f}%)")
    print(f"  买方 {len(bids)}档  卖方 {len(asks)}档", end="")
    if dt_ms:
        print(f"  {dt_ms:.1f}ms", end="")
    print()
    print(f"{'='*60}")

    # 卖方：升序显示，从卖一（最低卖价）开始
    print(f"\n  卖方:")
    for r in asks[:8]:
        print(f"    {r.price:10.2f}  {r.vol:10d}")
    if len(asks) > 8:
        print(f"    {'...':>10s}  ({len(asks)}档)")

    print(f"\n  ─── {bids[0].price:.2f} / {asks[0].price:.2f} ───")

    # 买方：降序显示，从买一（最高买价）开始
    print(f"\n  买方:")
    for r in bids[:8]:
        print(f"    {r.price:10.2f}  {r.vol:10d}")
    if len(bids) > 8:
        print(f"    {'...':>10s}  ({len(bids)}档)")

def monitor(interval=1.0):
    print(f"\n  持续监控 (间隔{interval}s, Ctrl+C退出)\n")

    bid_info, ask_info = load_cache()
    if bid_info and ask_info and validate_cache(bid_info, ask_info):
        print("[+] 缓存有效")
    else:
        bid_info, ask_info, _ = fast_discover()
        if not bid_info or not ask_info:
            print("[!] 发现失败")
            return
        save_cache(bid_info, ask_info)

    t0 = time.perf_counter()
    bids = read_side(*bid_info)
    asks = read_side(*ask_info)
    dt = (time.perf_counter() - t0) * 1000
    print_book(bids, asks, dt)

    frame = 0
    rescan = 0
    print(f"\n[*] 监控中...\n")
    try:
        while True:
            frame += 1
            time.sleep(interval)
            t0 = time.perf_counter()

            if rescan >= 120:
                if not validate_cache(bid_info, ask_info):
                    print(f"\n[!] 重新发现...")
                    bid_info, ask_info, _ = fast_discover()
                    if bid_info and ask_info:
                        save_cache(bid_info, ask_info)
                rescan = 0

            bids = read_side(*bid_info)
            asks = read_side(*ask_info)
            dt = (time.perf_counter() - t0) * 1000

            if bids and asks:
                bids.sort(key=lambda r: -r.price)
                asks.sort(key=lambda r:  r.price)
                sp = asks[0].price - bids[0].price
                print(f"[{frame:4d}] {dt:5.1f}ms | "
                      f"买={bids[0].price:.2f}×{bids[0].vol:<6d} "
                      f"卖={asks[0].price:.2f}×{asks[0].vol:<6d} "
                      f"差={sp:.4f} "
                      f"B{len(bids):3d} A{len(asks):3d}")
            else:
                print(f"[{frame:4d}] {dt:5.1f}ms | 读取失败")

            rescan += 1
    except KeyboardInterrupt:
        print(f"\n[*] 已停止 ({frame}帧)")

"""
ax_realtime_v15.py — 自动识别股票名称 + 自动发现买卖方 + 价格去重
用法:
  ax_realtime_v15.py                      # 全自动（识别窗口+发现价格）
  ax_realtime_v15.py --monitor 1          # 全自动 + 持续监控
  ax_realtime_v15.py --stock 600030       # 手动指定股票代码
  ax_realtime_v15.py --bid1 27.28 --ask1 27.30  # 手动指定价格
  ax_realtime_v15.py --clear              # 清除缓存重新发现
"""

import struct, time, sys, ctypes, warnings, json, os, re, argparse
from dataclasses import dataclass
from typing import List, Optional, Tuple
from ctypes import wintypes

import pymem
import pymem.process
import numpy as np

warnings.filterwarnings('ignore')

PROCESS    = "AX.exe"
REC_SIZE   = 20
REC_FMT    = "<fIIII"
CACHE_FILE = os.path.join(os.path.expanduser("~"), "ax_cache.json")

# ===== 全局状态（自动检测或命令行指定） =====
STOCK_CODE = None
STOCK_NAME = None
BID1       = None
ASK1       = None
PRICE_LO   = 5.0       # 初始宽范围，发现后自动收紧
PRICE_HI   = 500.0

@dataclass
class Rec:
    price: float
    vol: int
    cnt: int
    status: int
    seq: int
    offset: int = 0

pm  = pymem.Pymem(PROCESS)
mod = pymem.process.module_from_name(pm.process_handle, PROCESS)
print(f"[+] {PROCESS} PID={pm.process_id}")

kernel32 = ctypes.windll.kernel32

# ===== 内存区域枚举 =====
class MBI(ctypes.Structure):
    _fields_ = [
        ("BaseAddress",       ctypes.c_void_p),
        ("AllocationBase",    ctypes.c_void_p),
        ("AllocationProtect", ctypes.c_ulong),
        ("RegionSize",        ctypes.c_size_t),
        ("State",             ctypes.c_ulong),
        ("Protect",           ctypes.c_ulong),
        ("Type",              ctypes.c_ulong),
    ]

def enum_regions():
    regions = []
    mbi = MBI()
    cur = ctypes.c_size_t(0)
    while cur.value < 0x7FFFFFFFFFFF:
        ret = kernel32.VirtualQueryEx(
            pm.process_handle, ctypes.c_void_p(cur.value),
            ctypes.byref(mbi), ctypes.sizeof(mbi))
        if ret == 0:
            break
        base = mbi.BaseAddress
        sz   = mbi.RegionSize
        if base is None or sz == 0:
            cur.value += 0x1000; continue
        if mbi.State == 0x1000:
            prot = mbi.Protect & 0xFF
            if prot in (2, 4, 6, 8, 0x20, 0x40, 0x60, 0x80):
                if sz >= REC_SIZE * 10:
                    regions.append((base, sz))
        nxt = base + sz
        if nxt <= cur.value:
            cur.value += 0x1000; continue
        cur.value = nxt
    return regions

# ===== 窗口标题自动识别股票 =====
def detect_from_window():
    """从 AX.exe 窗口标题识别股票名称和代码"""
    global STOCK_CODE, STOCK_NAME

    titles = []
    try:
        EnumWindows = ctypes.windll.user32.EnumWindows
        GetWindowTextW = ctypes.windll.user32.GetWindowTextW
        GetWindowThreadProcessId = ctypes.windll.user32.GetWindowThreadProcessId
        IsWindowVisible = ctypes.windll.user32.IsWindowVisible

        WNDENUMPROC = ctypes.WINFUNCTYPE(
            wintypes.BOOL, wintypes.HWND, wintypes.LPARAM)

        def callback(hwnd, _):
            if not IsWindowVisible(hwnd):
                return True
            pid = wintypes.DWORD()
            GetWindowThreadProcessId(hwnd, ctypes.byref(pid))
            if pid.value == pm.process_id:
                buf = ctypes.create_unicode_buffer(512)
                GetWindowTextW(hwnd, buf, 512)
                title = buf.value.strip()
                if title and len(title) > 2:
                    titles.append(title)
            return True

        EnumWindows(WNDENUMPROC(callback), 0)
    except Exception:
        pass

    for title in titles:
        # 提取股票代码（5-6位数字）
        m = re.search(r'(\d{5,6})', title)
        if m and not STOCK_CODE:
            STOCK_CODE = m.group(1)
        # 提取股票名称（破折号/括号之前）
        name = re.split(r'[-—–(（]', title)[0].strip()
        if name and len(name) >= 2 and not STOCK_NAME:
            STOCK_NAME = name

    return titles

# ===== 数据读取（带价格去重） =====
def read_side(base, size, align):
    """读取一侧订单簿数据，按价格去重保留 seq 最大的记录"""
    try:
        raw = pm.read_bytes(base, size)
    except:
        return []
    recs = []
    pos = align
    while pos + REC_SIZE <= len(raw):
        try:
            p, v, c, s, q = struct.unpack_from(REC_FMT, raw, pos)
        except:
            break
        if PRICE_LO <= p <= PRICE_HI and 0 < q < 50000:
            recs.append(Rec(round(p, 4), v, c, s, q, base + pos))
        pos += REC_SIZE
    # 按价格去重：每个价格只保留 seq 最大的那条
    by_price = {}
    for r in recs:
        if r.price not in by_price or r.seq > by_price[r.price].seq:
            by_price[r.price] = r
    return list(by_price.values())

# ===== 区域对齐扫描 =====
def scan_region(base, size):
    """扫描一个区域的所有对齐偏移，返回评分结果"""
    try:
        raw = pm.read_bytes(base, size)
    except:
        return []

    results = []
    for a in range(REC_SIZE):
        recs = []
        pos = a
        while pos + REC_SIZE <= len(raw):
            try:
                p, v, c, s, q = struct.unpack_from(REC_FMT, raw, pos)
            except:
                break
            if PRICE_LO <= p <= PRICE_HI and 0 < q < 50000:
                recs.append(Rec(round(p, 4), v, c, s, q, base + pos))
            pos += REC_SIZE

        if len(recs) < 50:
            continue

        by_seq = sorted(recs, key=lambda r: r.seq)
        ps = [r.price for r in by_seq]
        asc  = sum(1 for i in range(1, len(ps)) if ps[i] >= ps[i-1])
        desc = sum(1 for i in range(1, len(ps)) if ps[i] <= ps[i-1])
        side = "bid" if desc >= asc else "ask"
        mono = max(asc, desc) / (len(ps) - 1) if len(ps) > 1 else 0

        density = len(recs) / size
        seqs = sorted(set(r.seq for r in recs))
        seq_cont = (sum(1 for i in range(1, len(seqs))
                       if seqs[i] - seqs[i-1] <= 2) / (len(seqs) - 1)
                   if len(seqs) >= 2 else 0)

        score = (min(len(recs) / 150, 1.0) * 30 +
                 mono * 25 + seq_cont * 25 +
                 min(density / 0.01, 1.0) * 20)

        results.append((score, base, size, a, side, len(recs), mono))

    return results

# ===== 自动发现（不需要预知价格） =====
def auto_discover():
    """自动发现：识别股票 → 扫描区域 → 确定买卖方和价格范围"""
    global BID1, ASK1, PRICE_LO, PRICE_HI

    t0 = time.perf_counter()

    # 1. 窗口识别
    print(f"\n[*] 识别股票...")
    titles = detect_from_window()
    if titles:
        print(f"    窗口: {titles[0]}")
    if STOCK_NAME or STOCK_CODE:
        label = f"{STOCK_NAME or '未知'} ({STOCK_CODE or '未知'})"
        print(f"[+] 股票: {label} [自动识别]")
    else:
        print(f"[!] 未从窗口识别到股票信息")

    # 2. 枚举区域
    print(f"\n[*] 枚举区域...")
    regions = enum_regions()
    print(f"[+] {len(regions)} 个区域")

    small = [(b, s) for b, s in regions if 1000 <= s <= 1024 * 1024]
    print(f"    候选区域 (1KB~1MB): {len(small)} 个")

    # 3. 广泛扫描（宽价格范围）
    old_lo, old_hi = PRICE_LO, PRICE_HI
    PRICE_LO, PRICE_HI = 5.0, 500.0

    print(f"\n[*] 广泛扫描价格区间 (5~500)...")
    candidates = []
    scanned = 0

    for base, size in small:
        try:
            raw = pm.read_bytes(base, size)
        except:
            continue
        scanned += 1

        # 快速过滤：检查价格类浮点数密度
        arr = np.frombuffer(raw, dtype='<f4')
        mask = np.isfinite(arr) & (arr >= 5.0) & (arr <= 500.0)
        count = int(np.sum(mask))
        if count < 80:
            continue

        results = scan_region(base, size)
        candidates.extend(results)

    dt = (time.perf_counter() - t0) * 1000
    candidates.sort(key=lambda x: -x[0])

    print(f"[+] 扫描 {scanned} 个区域, {len(candidates)} 个候选 ({dt:.0f}ms)\n")

    for i, (sc, base, sz, a, side, n, mono) in enumerate(candidates[:15]):
        print(f"  [{i:2d}] {sc:5.1f} {side:3s} 0x{base:X} a={a:2d} "
              f"{n:4d}条 sz={sz//1024}KB mono={mono:.2f}")

    # 4. 选择最佳买卖方区域
    bid_info = None
    ask_info = None
    for sc, base, sz, a, side, n, mono in candidates:
        if side == "bid" and bid_info is None:
            bid_info = (base, sz, a)
        elif side == "ask" and ask_info is None:
            ask_info = (base, sz, a)
        if bid_info and ask_info:
            break

    if not bid_info or not ask_info:
        PRICE_LO, PRICE_HI = old_lo, old_hi
        return None, None, dt

    # 5. 读取数据，确定实际价格范围
    bids = read_side(*bid_info)
    asks = read_side(*ask_info)

    if bids and asks:
        bids.sort(key=lambda r: -r.price)
        asks.sort(key=lambda r: r.price)
        BID1 = bids[0].price
        ASK1 = asks[0].price

        all_prices = [r.price for r in bids + asks]
        lo = min(all_prices)
        hi = max(all_prices)
        span = hi - lo
        PRICE_LO = round(lo - span * 0.1, 2)
        PRICE_HI = round(hi + span * 0.1, 2)

        print(f"\n[+] 买一={BID1:.4f}  卖一={ASK1:.4f}")
        print(f"[+] 价格范围: {PRICE_LO:.2f} ~ {PRICE_HI:.2f}")
    else:
        PRICE_LO, PRICE_HI = old_lo, old_hi

    return bid_info, ask_info, dt

# ===== 快速发现（已知价格时使用） =====
def fast_discover():
    """使用已知 BID1/ASK1 做精确 numpy 快速筛选"""
    t0 = time.perf_counter()

    print(f"\n[*] 枚举区域...")
    regions = enum_regions()
    print(f"[+] {len(regions)} 个区域")

    small = [(b, s) for b, s in regions if 1000 <= s <= 1024 * 1024]
    print(f"    候选区域 (1KB~1MB): {len(small)} 个")

    tol = max(BID1, ASK1) * 0.05
    print(f"\n[*] 精确搜索 BID1={BID1:.4f} ASK1={ASK1:.4f} ±{tol:.2f}...")

    candidates = []
    scanned = 0
    for base, size in small:
        try:
            raw = pm.read_bytes(base, size)
        except:
            continue
        scanned += 1

        arr = np.frombuffer(raw, dtype='<f4')
        mask = (np.isfinite(arr) &
                ((np.abs(arr - BID1) < tol) | (np.abs(arr - ASK1) < tol)))
        if not np.any(mask):
            continue

        results = scan_region(base, size)
        candidates.extend(results)

    dt = (time.perf_counter() - t0) * 1000
    candidates.sort(key=lambda x: -x[0])

    print(f"[+] 扫描 {scanned} 个区域, {len(candidates)} 个候选 ({dt:.0f}ms)\n")

    for i, (sc, base, sz, a, side, n, mono) in enumerate(candidates[:15]):
        print(f"  [{i:2d}] {sc:5.1f} {side:3s} 0x{base:X} a={a:2d} "
              f"{n:4d}条 sz={sz//1024}KB mono={mono:.2f}")

    bid_info = None
    ask_info = None
    for sc, base, sz, a, side, n, mono in candidates:
        if side == "bid" and bid_info is None:
            bid_info = (base, sz, a)
        elif side == "ask" and ask_info is None:
            ask_info = (base, sz, a)
        if bid_info and ask_info:
            break

    return bid_info, ask_info, dt

# ===== 缓存管理 =====
def save_cache(bid_info, ask_info):
    data = {
        "bid": list(bid_info),
        "ask": list(ask_info),
        "stock_code": STOCK_CODE,
        "stock_name": STOCK_NAME,
        "bid1": BID1,
        "ask1": ASK1,
        "price_lo": PRICE_LO,
        "price_hi": PRICE_HI,
        "time": time.time()
    }
    with open(CACHE_FILE, 'w') as f:
        json.dump(data, f)
    print(f"[+] 缓存已保存")

def load_cache():
    """加载缓存并恢复全局状态"""
    global STOCK_CODE, STOCK_NAME, BID1, ASK1, PRICE_LO, PRICE_HI

    if not os.path.exists(CACHE_FILE):
        return None, None
    try:
        with open(CACHE_FILE) as f:
            data = json.load(f)

        if data.get("stock_code"):
            STOCK_CODE = data["stock_code"]
        if data.get("stock_name"):
            STOCK_NAME = data["stock_name"]
        if data.get("bid1") is not None:
            BID1 = data["bid1"]
        if data.get("ask1") is not None:
            ASK1 = data["ask1"]
        if data.get("price_lo") is not None:
            PRICE_LO = data["price_lo"]
        if data.get("price_hi") is not None:
            PRICE_HI = data["price_hi"]

        return tuple(data["bid"]), tuple(data["ask"])
    except:
        return None, None

def validate_cache(bid_info, ask_info):
    """验证缓存数据是否仍然有效"""
    bids = read_side(*bid_info)
    asks = read_side(*ask_info)
    if len(bids) < 50 or len(asks) < 50:
        return False
    bids.sort(key=lambda r: -r.price)
    asks.sort(key=lambda r: r.price)
    sp = asks[0].price - bids[0].price
    return 0 < sp < 1.0

# ===== 显示 =====
def print_book(bids, asks, dt_ms=0):
    if not bids or not asks:
        print(f"[!] 数据不完整")
        return
    bids.sort(key=lambda r: -r.price)
    asks.sort(key=lambda r: r.price)
    sp = asks[0].price - bids[0].price
    md = (asks[0].price + bids[0].price) / 2

    label = f"{STOCK_NAME or ''} {STOCK_CODE or ''}".strip() or "未知"

    print(f"\n{'='*60}")
    print(f"  {label}  买一={bids[0].price:.4f} 卖一={asks[0].price:.4f} "
          f"差={sp:.4f}({sp/md*100:.3f}%)")
    print(f"  买方 {len(bids)}档  卖方 {len(asks)}档", end="")
    if dt_ms:
        print(f"  {dt_ms:.1f}ms", end="")
    print()
    print(f"{'='*60}")

    print(f"\n  卖方:")
    for r in asks[:8]:
        print(f"    {r.price:10.4f}  {r.vol:10d}")
    if len(asks) > 8:
        print(f"    {'...':>10s}  ({len(asks)}档)")

    print(f"\n  ─── {bids[0].price:.4f} / {asks[0].price:.4f} ───")

    print(f"\n  买方:")
    for r in bids[:8]:
        print(f"    {r.price:10.4f}  {r.vol:10d}")
    if len(bids) > 8:
        print(f"    {'...':>10s}  ({len(bids)}档)")

# ===== 持续监控 =====
def monitor(interval=1.0):
    print(f"\n  持续监控 (间隔{interval}s, Ctrl+C退出)\n")

    bid_info, ask_info = load_cache()
    cached_code = STOCK_CODE  # 保存缓存中的股票代码

    # 检测当前股票
    detect_from_window()

    stock_changed = (cached_code and STOCK_CODE and cached_code != STOCK_CODE)

    if bid_info and ask_info and not stock_changed and validate_cache(bid_info, ask_info):
        print("[+] 缓存有效")
    else:
        if stock_changed:
            print(f"[!] 股票变更: {cached_code} → {STOCK_CODE}")
        if BID1 and ASK1 and not stock_changed:
            bid_info, ask_info, _ = fast_discover()
        else:
            bid_info, ask_info, _ = auto_discover()
        if not bid_info or not ask_info:
            print("[!] 发现失败")
            return
        save_cache(bid_info, ask_info)

    t0 = time.perf_counter()
    bids = read_side(*bid_info)
    asks = read_side(*ask_info)

1	买卖方配对错误（来自不同股票）	致命	auto_discover 分别选最高分 bid/ask 区域，未验证是否同源
