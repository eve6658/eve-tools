千档盘口内存抓取工具 - v10 诊断版
核心改动：

import pymem
import struct
import time
import ctypes
import ctypes.wintypes
import sys
import os
from collections import Counter

try:
    import numpy as np
    HAS_NUMPY = True
except ImportError:
    HAS_NUMPY = False

PROCESS_NAME = "AX.exe"
RECORD_SIZE = 20
PRICE_MIN = 50.0
PRICE_MAX = 200.0
VOL_MAX = 10_000_000
CNT_MAX = 1_000_000
MAX_SEQ = 2000
MAX_REGION_SIZE = 80 * 1024 * 1024
MIN_RUN_LEN = 10
MEM_COMMIT = 0x1000
PAGE_NOACCESS = 0x01
PAGE_GUARD = 0x100


class MEMORY_BASIC_INFORMATION(ctypes.Structure):
    _fields_ = [
        ("BaseAddress", ctypes.c_void_p),
        ("AllocationBase", ctypes.c_void_p),
        ("AllocationProtect", ctypes.wintypes.DWORD),
        ("RegionSize", ctypes.c_size_t),
        ("State", ctypes.wintypes.DWORD),
        ("Protect", ctypes.wintypes.DWORD),
        ("Type", ctypes.wintypes.DWORD),
    ]


def si(val):
    return int(val) if val is not None else 0


def get_readable_regions(pm):
    regions = []
    addr = 0
    mbi = MEMORY_BASIC_INFORMATION()
    mbi_size = ctypes.sizeof(mbi)
    kernel32 = ctypes.windll.kernel32
    while True:
        ret = kernel32.VirtualQueryEx(
            pm.process_handle, ctypes.c_void_p(addr),
            ctypes.byref(mbi), mbi_size
        )
        if not ret:
            break
        base = si(mbi.BaseAddress)
        size = si(mbi.RegionSize)
        state = si(mbi.State)
        protect = si(mbi.Protect)
        if (state == MEM_COMMIT and size > 0
                and protect != PAGE_NOACCESS
                and not (protect & PAGE_GUARD)):
            regions.append((base, size))
        nxt = base + size
        if nxt <= addr or nxt >= 0x7FFFFFFFFFFF:
            break
        addr = nxt
    return regions


def numpy_find_candidates(raw_bytes):
    if len(raw_bytes) < RECORD_SIZE * 3:
        return []
    buf = bytes(raw_bytes)
    if len(buf) % 4 != 0:
        buf = buf[: len(buf) & ~3]
    all_u32 = np.frombuffer(buf, dtype="<u4")
    all_f32 = np.frombuffer(buf, dtype="<f4")
    n = len(all_u32) - 14
    if n <= 0:
        return []
    p0 = all_f32[0:n]
    v0 = all_u32[1:n + 1]
    s0 = all_u32[4:n + 4]
    p1 = all_f32[5:n + 5]
    v1 = all_u32[6:n + 6]
    s1 = all_u32[9:n + 9]
    p2 = all_f32[10:n + 10]
    v2 = all_u32[11:n + 11]
    s2 = all_u32[14:n + 14]
    mask = (
        (p0 >= PRICE_MIN) & (p0 <= PRICE_MAX) & (v0 > 0) & (v0 < VOL_MAX) & (s0 < MAX_SEQ)
        & (p1 >= PRICE_MIN) & (p1 <= PRICE_MAX) & (v1 > 0) & (v1 < VOL_MAX) & (s1 < MAX_SEQ)
        & (p2 >= PRICE_MIN) & (p2 <= PRICE_MAX) & (v2 > 0) & (v2 < VOL_MAX) & (s2 < MAX_SEQ)
        & (s1 == s0 + 1)
        & (s2 == s0 + 2)
    )
    indices = np.where(mask)[0]
    results = []
    for idx in indices[:50]:
        i = int(idx)
        results.append((i * 4, int(s0[i])))
    return results


def fallback_find_candidates(raw_bytes):
    results = []
    limit = len(raw_bytes) - RECORD_SIZE * 3
    for off in range(0, limit, 4):
        p0 = struct.unpack_from("<f", raw_bytes, off)[0]
        v0, _, _, s0 = struct.unpack_from("<4I", raw_bytes, off + 4)
        if not (PRICE_MIN <= p0 <= PRICE_MAX and 0 < v0 < VOL_MAX and s0 < MAX_SEQ):
            continue
        p1 = struct.unpack_from("<f", raw_bytes, off + RECORD_SIZE)[0]
        v1, _, _, s1 = struct.unpack_from("<4I", raw_bytes, off + RECORD_SIZE + 4)
        if not (PRICE_MIN <= p1 <= PRICE_MAX and 0 < v1 < VOL_MAX and s1 == s0 + 1):
            continue
        p2 = struct.unpack_from("<f", raw_bytes, off + RECORD_SIZE * 2)[0]
        v2, _, _, s2 = struct.unpack_from("<4I", raw_bytes, off + RECORD_SIZE * 2 + 4)
        if not (PRICE_MIN <= p2 <= PRICE_MAX and 0 < v2 < VOL_MAX and s2 == s0 + 2):
            continue
        results.append((off, s0))
        if len(results) >= 50:
            break
    return results


def extract_records(raw_bytes, base_addr, alignment):
    records = []
    for i in range(alignment, len(raw_bytes) - RECORD_SIZE + 1, RECORD_SIZE):
        price = struct.unpack_from("<f", raw_bytes, i)[0]
        vol, cnt, status, seq = struct.unpack_from("<4I", raw_bytes, i + 4)
        if (PRICE_MIN <= price <= PRICE_MAX
                and 0 < vol < VOL_MAX
                and cnt < CNT_MAX
                and 0 <= seq <= MAX_SEQ):
            records.append((price, vol, cnt, status, seq, base_addr + i))
    return records


def find_consecutive_runs(records, min_len=MIN_RUN_LEN):
    if len(records) < min_len:
        return []
    runs = []
    cur = [records[0]]
    for i in range(1, len(records)):
        if records[i][4] == cur[-1][4] + 1:
            cur.append(records[i])
        else:
            if len(cur) >= min_len:
                runs.append(cur)
            cur = [records[i]]
    if len(cur) >= min_len:
        runs.append(cur)
    return runs


def validate_run(run):
    if len(run) < 3:
        return False
    prices = [r[0] for r in run]
    increasing = all(prices[i] <= prices[i + 1] for i in range(len(prices) - 1))
    decreasing = all(prices[i] >= prices[i + 1] for i in range(len(prices) - 1))
    return increasing or decreasing


def deduplicate(records):
    seen = set()
    unique = []
    for r in records:
        key = (round(r[0], 4), r[1], r[4])
        if key not in seen:
            seen.add(key)
            unique.append(r)
    return unique


def split_multi_stock_runs(run):
    """在显著价格向上跳变处拆分序列（多股票数据混合检测）"""
    if len(run) < 30:
        return [run]

    prices = [r[0] for r in run]
    n = len(prices)

    # 计算典型间距
    gaps = sorted(abs(prices[i] - prices[i - 1]) for i in range(1, n))
    median_gap = gaps[len(gaps) // 2] if gaps else 0.10

    # 阈值：10倍中位间距或至少2.0
    threshold = max(median_gap * 10, 2.0)

    # 找显著向上跳变
    split_points = [0]
    for i in range(1, n):
        if prices[i] > prices[i - 1] + threshold:
            split_points.append(i)
    split_points.append(n)

    sub_runs = []
    for i in range(len(split_points) - 1):
        start = split_points[i]
        end = split_points[i + 1]
        if end - start >= 20:
            sub_runs.append(run[start:end])

    return sub_runs if sub_runs else [run]


def find_best_split(run):
    """在价格递减序列中找卖方/买方分界点
    条件：分界点之前所有价格 > 分界点之后所有价格
    选择：余额比最优的分界点
    """
    n = len(run)
    if n < 20:
        return None

    prices = [r[0] for r in run]

    # 预计算前缀最小值和后缀最大值
    prefix_min = [0.0] * n
    suffix_max = [0.0] * n
    prefix_min[0] = prices[0]
    for i in range(1, n):
        prefix_min[i] = min(prefix_min[i - 1], prices[i])
    suffix_max[n - 1] = prices[n - 1]
    for i in range(n - 2, -1, -1):
        suffix_max[i] = max(suffix_max[i + 1], prices[i])

    best_idx = None
    best_balance = -1

    for i in range(10, n - 10):
        ask_min = prefix_min[i - 1]  # run[0:i] 的最小值
        bid_max = suffix_max[i]      # run[i:n] 的最大值

        if ask_min <= bid_max:
            continue

        spread = ask_min - bid_max
        if spread > 5.0:
            continue

        balance = min(i, n - i) / max(i, n - i)

        if balance > best_balance:
            best_balance = balance
            best_idx = i

    return best_idx


def build_order_book(all_runs):
    """从所有连续序列中构建最佳订单簿"""
    candidates = []

    # 先拆分多股票混合序列
    all_sub_runs = []
    for run in all_runs:
        all_sub_runs.extend(split_multi_stock_runs(run))

    # ===== DIAG: split_multi_stock_runs 结果 =====
    print(f"\n===== DIAG: split_multi_stock_runs 结果 =====")
    print(f"  输入: {len(all_runs)} 个序列, 输出: {len(all_sub_runs)} 个子序列")
    for i, run in enumerate(all_sub_runs):
        seqs = [r[4] for r in run]
        prices = [r[0] for r in run]
        print(f"    sub[{i}]: 长度={len(run)}, "
              f"seq范围={min(seqs)}-{max(seqs)}, "
              f"价格范围={min(prices):.2f}-{max(prices):.2f}")
    # ===== DIAG END =====

    # 策略1：拆分单个递减序列为卖方+买方
    for run in all_sub_runs:
        if len(run) < 30:
            continue
        prices = [r[0] for r in run]
        n = len(prices)
        down = sum(1 for i in range(n - 1) if prices[i + 1] < prices[i] - 0.001)
        if down < n * 0.3:
            continue  # 不是主要递减，跳过

        split = find_best_split(run)

        # ===== DIAG: 策略1 候选 =====
        print(f"\n===== DIAG: 策略1 候选 =====")
        print(f"  序列长度={len(run)}, 递减比例={down/n:.2f}, 分界点={split}")
        if split:
            print(f"  分界后: 卖方={split}档, 买方={len(run)-split}档")
            ask_prices = [r[0] for r in run[:split]]
            bid_prices = [r[0] for r in run[split:]]
            print(f"  卖方价格: {min(ask_prices):.2f} ~ {max(ask_prices):.2f}")
            print(f"  买方价格: {min(bid_prices):.2f} ~ {max(bid_prices):.2f}")
        # ===== DIAG END =====

        if split and 10 <= split <= len(run) - 10:
            asks = sorted(run[:split], key=lambda r: r[0])
            bids = sorted(run[split:], key=lambda r: -r[0])
            if asks and bids and asks[0][0] > bids[0][0]:
                spread = asks[0][0] - bids[0][0]
                if 0 < spread < 5.0:
                    balance = min(len(asks), len(bids)) / max(1, max(len(asks), len(bids)))
                    if balance >= 0.1:
                        cache = [(run[0][5], len(run))]
                        candidates.append((asks, bids, balance, cache))

    # 策略2：配对递增序列（卖方）和递减序列（买方）
    asc_runs = []
    desc_runs = []
    for run in all_sub_runs:
        if len(run) < 10:
            continue
        prices = [r[0] for r in run]
        n = len(prices)
        up = sum(1 for i in range(n - 1) if prices[i + 1] > prices[i] + 0.001)
        down = sum(1 for i in range(n - 1) if prices[i + 1] < prices[i] - 0.001)
        if up > down:
            asc_runs.append(run)
        else:
            desc_runs.append(run)

    # ===== DIAG: 策略2 分类 =====
    print(f"\n===== DIAG: 策略2 分类 =====")
    print(f"  递增序列: {len(asc_runs)} 个, 递减序列: {len(desc_runs)} 个")
    for i, run in enumerate(asc_runs):
        prices = [r[0] for r in run]
        print(f"    asc[{i}]: 长度={len(run)}, 价格={min(prices):.2f}~{max(prices):.2f}")
    for i, run in enumerate(desc_runs):
        prices = [r[0] for r in run]
        print(f"    desc[{i}]: 长度={len(run)}, 价格={min(prices):.2f}~{max(prices):.2f}")
    # ===== DIAG END =====

    for a_run in asc_runs:
        a_min = min(r[0] for r in a_run)
        for b_run in desc_runs:
            b_max = max(r[0] for r in b_run)
            if a_min > b_max:
                spread = a_min - b_max
                if 0 < spread < 5.0:
                    asks = sorted(a_run, key=lambda r: r[0])
                    bids = sorted(b_run, key=lambda r: -r[0])
                    balance = min(len(asks), len(bids)) / max(1, max(len(asks), len(bids)))
                    if balance >= 0.1:
                        cache = [
                            (a_run[0][5], len(a_run)),
                            (b_run[0][5], len(b_run)),
                        ]
                        candidates.append((asks, bids, balance, cache))

    # ===== DIAG: 最终候选 =====
    print(f"\n===== DIAG: 最终候选数 =====")
    print(f"  candidates: {len(candidates)}")
    for i, c in enumerate(candidates):
        print(f"    [{i}] 卖方={len(c[0])}档 买方={len(c[1])}档 balance={c[2]:.4f}")
    # ===== DIAG END =====

    if not candidates:
        return None

    # 选最优候选：余额比最高
    best = max(candidates, key=lambda c: c[2])

    # ===== DIAG: 最终选择 =====
    print(f"\n===== DIAG: 最终选择 =====")
    print(f"  卖方={len(best[0])}档, 买方={len(best[1])}档")
    print(f"  卖方价格: {min(r[0] for r in best[0]):.2f} ~ {max(r[0] for r in best[0]):.2f}")
    print(f"  买方价格: {min(r[0] for r in best[1]):.2f} ~ {max(r[0] for r in best[1]):.2f}")
    # ===== DIAG END =====

    return best[0], best[1], best[3]


def full_scan(pm, regions):
    all_runs = []
    total = len(regions)

    # ===== DIAG: 扫描概览 =====
    print(f"\n===== DIAG: full_scan 开始 =====")
    print(f"  总区域数: {total}")
    # ===== DIAG END =====

    region_count = 0
    skip_count = 0
    candidate_count = 0
    run_count = 0

    for idx, (base, size) in enumerate(regions):
        if size > MAX_REGION_SIZE:
            skip_count += 1
            continue
        if idx % 300 == 0 and idx > 0:
            print(f"  扫描进度: {idx}/{total} ...")
        try:
            raw = pm.read_bytes(base, size)
        except Exception:
            continue

        region_count += 1
        candidates = (numpy_find_candidates(raw) if HAS_NUMPY
                      else fallback_find_candidates(raw))
        if not candidates:
            continue

        candidate_count += 1
        align_counts = Counter(off % RECORD_SIZE for off, _ in candidates)
        best_align = align_counts.most_common(1)[0][0]
        records = extract_records(raw, base, best_align)
        runs = find_consecutive_runs(records)

        best_run = None
        for run in runs:
            if validate_run(run):
                if best_run is None or len(run) > len(best_run):
                    best_run = run

        if best_run and len(best_run) >= MIN_RUN_LEN:
            run_count += 1
            all_runs.append(best_run)

    # ===== DIAG: full_scan 汇总 =====
    print(f"\n===== DIAG: full_scan 汇总 =====")
    print(f"  跳过大区域: {skip_count}")
    print(f"  实际读取区域: {region_count}")
    print(f"  有候选记录的区域: {candidate_count}")
    print(f"  找到的连续序列: {run_count}")
    print(f"  全局共找到 {len(all_runs)} 个连续序列:")
    for i, run in enumerate(all_runs):
        seqs = [r[4] for r in run]
        prices = [r[0] for r in run]
        print(f"    run[{i}]: 长度={len(run)}, "
              f"seq范围={min(seqs)}-{max(seqs)}, "
              f"价格范围={min(prices):.2f}-{max(prices):.2f}")
    # ===== DIAG END =====

    if not all_runs:
        return None

    result = build_order_book(all_runs)
    if not result:
        return None

    asks, bids, cache_blocks = result
    return asks, bids, cache_blocks


def cached_read(pm, cache_blocks):
    all_runs = []
    for addr, count in cache_blocks:
        try:
            raw = pm.read_bytes(addr, count * RECORD_SIZE)
        except Exception:
            return None
        records = []
        for i in range(count):
            off = i * RECORD_SIZE
            price = struct.unpack_from("<f", raw, off)[0]
            vol, cnt, status, seq = struct.unpack_from("<4I", raw, off + 4)
            if (PRICE_MIN <= price <= PRICE_MAX
                    and 0 < vol < VOL_MAX
                    and cnt < CNT_MAX
                    and 0 <= seq <= MAX_SEQ):
                records.append((price, vol, cnt, status, seq, addr + off))
        runs = find_consecutive_runs(records, min_len=5)
        all_runs.extend(runs)

    if not all_runs:
        return None

    result = build_order_book(all_runs)
    if not result:
        return None

    return result[0], result[1]


def display_books(asks, bids, scan_time, method, scan_count,
                  cache_hits, cache_misses):

    best_ask = asks[0][0]
    best_bid = bids[0][0]
    spread = best_ask - best_bid

    print("=" * 66)

    print(f"  {'-' * 46}")
    total_a = len(asks)
    show = min(25, total_a)
    for rank in range(show):
        i = total_a - 1 - rank
        p, v, c = asks[i][0], asks[i][1], asks[i][2]
        print(f"  {'卖' + str(i + 1):>6}  {p:>10.2f}  {v:>10}  {c:>8}")
    if total_a > show:
        print(f"  {'...':>6}  {'...':>10}  {'...':>10}  {'...':>8}")
        for i in range(3):
            p, v, c = asks[i][0], asks[i][1], asks[i][2]
            print(f"  {'卖' + str(i + 1):>6}  {p:>10.2f}  {v:>10}  {c:>8}")

    # 买方：买1=最高买价在顶部
    print(f"\n{'买方':^64}")
    print(f"  {'档位':>6}  {'价格':>10}  {'委托量':>10}  {'笔数':>8}")
    print(f"  {'-' * 46}")
    show = min(25, len(bids))
    for i in range(show):
        p, v, c = bids[i][0], bids[i][1], bids[i][2]
        print(f"  {'买' + str(i + 1):>6}  {p:>10.2f}  {v:>10}  {c:>8}")
    if len(bids) > show:
        print(f"  {'...':>6}  {'...':>10}  {'...':>10}  {'...':>8}")
        for i in range(len(bids) - 3, len(bids)):
            p, v, c = bids[i][0], bids[i][1], bids[i][2]
            print(f"  {'买' + str(i + 1):>6}  {p:>10.2f}  {v:>10}  {c:>8}")

    print(f"\n{'=' * 66}")
    print(f"  方法: {method}  |  扫描 {scan_count} 次"
          f"  |  命中 {cache_hits}  |  失效 {cache_misses}")
    print("=" * 66)


def main():
    print("正在连接 AX.exe ...")
    try:
        pm = pymem.Pymem(PROCESS_NAME)
    except Exception as e:
        print(f"无法连接: {e}")
        sys.exit(1)
    print(f"已连接 PID: {pm.process_id}")
    print("正在枚举内存区域 ...")
    regions = get_readable_regions(pm)

class MEMORY_BASIC_INFORMATION(ctypes.Structure):
    _fields_ = [
        ("BaseAddress", ctypes.c_void_p),
        ("AllocationBase", ctypes.c_void_p),
        ("AllocationProtect", ctypes.wintypes.DWORD),
        ("RegionSize", ctypes.c_size_t),

def si(val):

def get_readable_regions(pm):
    regions = []
    addr = 0
    mbi = MEMORY_BASIC_INFORMATION()
    mbi_size = ctypes.sizeof(mbi)
    kernel32 = ctypes.windll.kernel32
    while True:
        ret = kernel32.VirtualQueryEx(
            pm.process_handle, ctypes.c_void_p(addr),
            ctypes.byref(mbi), mbi_size
        )
        if not ret:
            break
        base = si(mbi.BaseAddress)
        size = si(mbi.RegionSize)
        state = si(mbi.State)
        protect = si(mbi.Protect)
        if (state == MEM_COMMIT and size > 0
                and protect != PAGE_NOACCESS
                and not (protect & PAGE_GUARD)):
            regions.append((base, size))
        nxt = base + size
        if nxt <= addr or nxt >= 0x7FFFFFFFFFFF:
            break
        addr = nxt
    return regions


def numpy_find_candidates(raw_bytes):
    if len(raw_bytes) < RECORD_SIZE * 3:
        return []
    buf = bytes(raw_bytes)
    if len(buf) % 4 != 0:
        buf = buf[: len(buf) & ~3]
    all_u32 = np.frombuffer(buf, dtype="<u4")
    all_f32 = np.frombuffer(buf, dtype="<f4")
    n = len(all_u32) - 14
    if n <= 0:
        return []
    p0 = all_f32[0:n]
    v0 = all_u32[1:n + 1]
    s0 = all_u32[4:n + 4]
    p1 = all_f32[5:n + 5]
    v1 = all_u32[6:n + 6]
    s1 = all_u32[9:n + 9]
    p2 = all_f32[10:n + 10]
    v2 = all_u32[11:n + 11]
    s2 = all_u32[14:n + 14]
    mask = (
        (p0 >= PRICE_MIN) & (p0 <= PRICE_MAX) & (v0 > 0) & (v0 < VOL_MAX) & (s0 < MAX_SEQ)
        & (p1 >= PRICE_MIN) & (p1 <= PRICE_MAX) & (v1 > 0) & (v1 < VOL_MAX) & (s1 < MAX_SEQ)
        & (p2 >= PRICE_MIN) & (p2 <= PRICE_MAX) & (v2 > 0) & (v2 < VOL_MAX) & (s2 < MAX_SEQ)
        & (s1 == s0 + 1)
        & (s2 == s0 + 2)
    )
    indices = np.where(mask)[0]
    results = []
    for idx in indices[:50]:
        i = int(idx)
        results.append((i * 4, int(s0[i])))
    return results


def fallback_find_candidates(raw_bytes):

def extract_records(raw_bytes, base_addr, alignment):
    records = []

def find_consecutive_runs(records, min_len=MIN_RUN_LEN):
    if len(records) < min_len:
        return []
    runs = []
    cur = [records[0]]
    for i in range(1, len(records)):
        if records[i][4] == cur[-1][4] + 1:
            cur.append(records[i])
        else:
            if len(cur) >= min_len:
                runs.append(cur)
            cur = [records[i]]
    if len(cur) >= min_len:
        runs.append(cur)
    return runs


def validate_run(run):
    if len(run) < 3:
        return False
    prices = [r[0] for r in run]
    increasing = all(prices[i] <= prices[i + 1] for i in range(len(prices) - 1))
    decreasing = all(prices[i] >= prices[i + 1] for i in range(len(prices) - 1))
    return increasing or decreasing


def deduplicate(records):
    seen = set()
    unique = []
    for r in records:
        key = (round(r[0], 4), r[1], r[4])
        if key not in seen:
            seen.add(key)
            unique.append(r)
    return unique


def split_multi_stock_runs(run):
    """在显著价格向上跳变处拆分序列（多股票数据混合检测）"""
    if len(run) < 30:
        return [run]

    prices = [r[0] for r in run]
    n = len(prices)

    # 计算典型间距
    gaps = sorted(abs(prices[i] - prices[i - 1]) for i in range(1, n))
    median_gap = gaps[len(gaps) // 2] if gaps else 0.10

    # 阈值：10倍中位间距或至少2.0
    threshold = max(median_gap * 10, 2.0)

    # 找显著向上跳变
    split_points = [0]
    for i in range(1, n):
        if prices[i] > prices[i - 1] + threshold:
            split_points.append(i)
    split_points.append(n)

    sub_runs = []
    for i in range(len(split_points) - 1):
        start = split_points[i]
        end = split_points[i + 1]
        if end - start >= 20:
            sub_runs.append(run[start:end])

    return sub_runs if sub_runs else [run]


def find_best_split(run):
    """在价格递减序列中找卖方/买方分界点
    条件：分界点之前所有价格 > 分界点之后所有价格
    选择：余额比最优的分界点
    """
    n = len(run)
    if n < 20:
        return None

    prices = [r[0] for r in run]

    # 预计算前缀最小值和后缀最大值
    prefix_min = [0.0] * n
    suffix_max = [0.0] * n
    prefix_min[0] = prices[0]
    for i in range(1, n):
        prefix_min[i] = min(prefix_min[i - 1], prices[i])
    suffix_max[n - 1] = prices[n - 1]
    for i in range(n - 2, -1, -1):
        suffix_max[i] = max(suffix_max[i + 1], prices[i])

    best_idx = None
    best_balance = -1

    # ===== DIAG: 分界点搜索过程 =====
    diag(f"  find_best_split: 序列长度={n}")
    diag(f"  价格范围: {prices[0]:.2f} ~ {prices[-1]:.2f}")
    valid_splits = []
    for i in range(10, n - 10):
        ask_min = prefix_min[i - 1]
        bid_max = suffix_max[i]

        if ask_min <= bid_max:
            continue

        spread = ask_min - bid_max
        if spread > 5.0:
            continue

        balance = min(i, n - i) / max(i, n - i)
        valid_splits.append((i, ask_min, bid_max, spread, balance))

        if balance > best_balance:
            best_balance = balance
            best_idx = i

    diag(f"  有效分界点数量: {len(valid_splits)}")
    if valid_splits:
        # 打印余额比最高的前5个
        top = sorted(valid_splits, key=lambda x: -x[4])[:5]
        for idx, (i, am, bm, sp, bal) in enumerate(top):
            diag(f"    top[{idx}]: pos={i}, 卖方min={am:.2f}, 买方max={bm:.2f}, "
                 f"spread={sp:.2f}, balance={bal:.4f}, "
                 f"卖方={i}档, 买方={n-i}档")
    # ===== DIAG END =====

    return best_idx


def build_order_book(all_runs):
    """从所有连续序列中构建最佳订单簿"""
    candidates = []

    # 先拆分多股票混合序列
    all_sub_runs = []
    for run in all_runs:
        all_sub_runs.extend(split_multi_stock_runs(run))

    # ===== DIAG: split_multi_stock_runs 结果 =====
    diag_flush()
    diag("DIAG: split_multi_stock_runs 结果")
    diag(f"  输入: {len(all_runs)} 个序列, 输出: {len(all_sub_runs)} 个子序列")
    for i, run in enumerate(all_sub_runs):
        seqs = [r[4] for r in run]
        prices = [r[0] for r in run]
        diag(f"  sub[{i}]: 长度={len(run)}, "

        # 打印前3条和后3条记录的详细信息
        diag(f"    前3条: {[(round(r[0],2), r[1], r[4]) for r in run[:3]]}")
        diag(f"    后3条: {[(round(r[0],2), r[1], r[4]) for r in run[-3:]]}")
    # ===== DIAG END =====

    # 策略1：拆分单个递减序列为卖方+买方
    for run_idx, run in enumerate(all_sub_runs):
        if len(run) < 30:
            continue
        prices = [r[0] for r in run]
        n = len(prices)
        down = sum(1 for i in range(n - 1) if prices[i + 1] < prices[i] - 0.001)
        if down < n * 0.3:
            continue  # 不是主要递减，跳过

        # ===== DIAG: 策略1 候选 =====
        diag_flush()
        diag(f"DIAG: 策略1 候选 sub[{run_idx}]")
        diag(f"  序列长度={len(run)}, 递减比例={down/n:.2f}")
        # ===== DIAG END =====

        split = find_best_split(run)

        # ===== DIAG: 分界结果 =====
        diag(f"  分界点={split}")
        if split:
            diag(f"  分界后: 卖方={split}档, 买方={len(run)-split}档")
            ask_prices = [r[0] for r in run[:split]]
            bid_prices = [r[0] for r in run[split:]]
            diag(f"  卖方价格: {min(ask_prices):.2f} ~ {max(ask_prices):.2f}")
            diag(f"  买方价格: {min(bid_prices):.2f} ~ {max(bid_prices):.2f}")
            diag(f"  卖方前3条seq: {[r[4] for r in run[:3]]}")
            diag(f"  卖方后3条seq: {[r[4] for r in run[split-3:split]]}")
            diag(f"  买方前3条seq: {[r[4] for r in run[split:split+3]]}")
            diag(f"  买方后3条seq: {[r[4] for r in run[-3:]]}")
        # ===== DIAG END =====

        if split and 10 <= split <= len(run) - 10:
            asks = sorted(run[:split], key=lambda r: r[0])
            bids = sorted(run[split:], key=lambda r: -r[0])
            if asks and bids and asks[0][0] > bids[0][0]:
                spread = asks[0][0] - bids[0][0]
                if 0 < spread < 5.0:
                    balance = min(len(asks), len(bids)) / max(1, max(len(asks), len(bids)))
                    if balance >= 0.1:
                        cache = [(run[0][5], len(run))]
                        candidates.append((asks, bids, balance, cache))
                        diag(f"  -> 策略1候选已添加, balance={balance:.4f}")
                    else:
                        diag(f"  -> 丢弃: balance={balance:.4f} < 0.1")
                else:
                    diag(f"  -> 丢弃: spread={spread:.2f} 不在 (0, 5.0)")
            else:
                if asks and bids:
                    diag(f"  -> 丢弃: 卖一={asks[0][0]:.2f} <= 买一={bids[0][0]:.2f}")
                else:
                    diag(f"  -> 丢弃: asks或bids为空")

    # 策略2：配对递增序列（卖方）和递减序列（买方）

    for run in all_sub_runs:
        if len(run) < 10:

        prices = [r[0] for r in run]
        n = len(prices)
        up = sum(1 for i in range(n - 1) if prices[i + 1] > prices[i] + 0.001)
        down = sum(1 for i in range(n - 1) if prices[i + 1] < prices[i] - 0.001)
        if up > down:
            asc_runs.append(run)
        else:
            desc_runs.append(run)

    # ===== DIAG: 策略2 分类 =====
    diag_flush()
    diag("DIAG: 策略2 分类")
    diag(f"  递增序列: {len(asc_runs)} 个, 递减序列: {len(desc_runs)} 个")
    for i, run in enumerate(asc_runs):
        prices = [r[0] for r in run]
        diag(f"  asc[{i}]: 长度={len(run)}, 价格={min(prices):.2f}~{max(prices):.2f}")
    for i, run in enumerate(desc_runs):
        prices = [r[0] for r in run]
        diag(f"  desc[{i}]: 长度={len(run)}, 价格={min(prices):.2f}~{max(prices):.2f}")
    # ===== DIAG END =====

    for a_run in asc_runs:
        a_min = min(r[0] for r in a_run)
        for b_run in desc_runs:
            b_max = max(r[0] for r in b_run)
            if a_min > b_max:
                spread = a_min - b_max
                if 0 < spread < 5.0:
                    asks = sorted(a_run, key=lambda r: r[0])
                    bids = sorted(b_run, key=lambda r: -r[0])
                    balance = min(len(asks), len(bids)) / max(1, max(len(asks), len(bids)))
                    if balance >= 0.1:

                            (a_run[0][5], len(a_run)),
                            (b_run[0][5], len(b_run)),
                        ]
                        candidates.append((asks, bids, balance, cache))
                        diag(f"  -> 策略2候选: 卖方={len(asks)}档 买方={len(bids)}档 "
                             f"balance={balance:.4f}")

    # ===== DIAG: 最终候选 =====
    diag_flush()
    diag("DIAG: 最终候选数")
    diag(f"  candidates: {len(candidates)}")
    for i, c in enumerate(candidates):
        diag(f"  [{i}] 卖方={len(c[0])}档 买方={len(c[1])}档 "
             f"balance={c[2]:.4f}")
        diag(f"    卖方价格: {min(r[0] for r in c[0]):.2f} ~ {max(r[0] for r in c[0]):.2f}")
        diag(f"    买方价格: {min(r[0] for r in c[1]):.2f} ~ {max(r[0] for r in c[1]):.2f}")

        diag("  没有候选，返回 None")
        return None

    # 选最优候选：余额比最高
    best = max(candidates, key=lambda c: c[2])

    # ===== DIAG: 最终选择 =====
    diag_flush()
    diag("DIAG: 最终选择")
    diag(f"  卖方={len(best[0])}档, 买方={len(best[1])}档")
    diag(f"  卖方价格: {min(r[0] for r in best[0]):.2f} ~ {max(r[0] for r in best[0]):.2f}")
    diag(f"  买方价格: {min(r[0] for r in best[1]):.2f} ~ {max(r[0] for r in best[1]):.2f}")
    diag(f"  卖方前5档(高价端):")
    asks_sorted = sorted(best[0], key=lambda r: -r[0])
    for i, r in enumerate(asks_sorted[:5]):
        diag(f"    卖{len(best[0])-i}: price={r[0]:.2f} vol={r[1]} cnt={r[2]} seq={r[4]}")
    diag(f"  卖方后5档(低价端):")
    asks_sorted_asc = sorted(best[0], key=lambda r: r[0])
    for i, r in enumerate(asks_sorted_asc[:5]):
        diag(f"    卖{i+1}: price={r[0]:.2f} vol={r[1]} cnt={r[2]} seq={r[4]}")
    diag(f"  买方前5档(高价端):")
    bids_sorted = sorted(best[1], key=lambda r: -r[0])
    for i, r in enumerate(bids_sorted[:5]):
        diag(f"    买{i+1}: price={r[0]:.2f} vol={r[1]} cnt={r[2]} seq={r[4]}")
    diag(f"  买方后5档(低价端):")
    bids_sorted_asc = sorted(best[1], key=lambda r: r[0])
    for i, r in enumerate(bids_sorted_asc[:5]):
        diag(f"    买{len(best[1])-i}: price={r[0]:.2f} vol={r[1]} cnt={r[2]} seq={r[4]}")
    # ===== DIAG END =====

    return best[0], best[1], best[3]


def full_scan(pm, regions):
    all_runs = []
    total = len(regions)

    # ===== DIAG: 扫描概览 =====
    diag_flush()
    diag("DIAG: full_scan 开始")
    diag(f"  总区域数: {total}")
    # ===== DIAG END =====

    region_count = 0
    skip_count = 0
    candidate_count = 0
    run_count = 0

    for idx, (base, size) in enumerate(regions):
        if size > MAX_REGION_SIZE:
            skip_count += 1
            continue
        if idx % 300 == 0 and idx > 0:
            diag(f"  扫描进度: {idx}/{total} ...")
        try:
            raw = pm.read_bytes(base, size)
        except Exception:
            continue

        region_count += 1
        candidates = (numpy_find_candidates(raw) if HAS_NUMPY
                      else fallback_find_candidates(raw))
        if not candidates:
            continue

        candidate_count += 1
        align_counts = Counter(off % RECORD_SIZE for off, _ in candidates)
        best_align = align_counts.most_common(1)[0][0]
        records = extract_records(raw, base, best_align)
        runs = find_consecutive_runs(records)

        best_run = None
        for run in runs:
            if validate_run(run):
                if best_run is None or len(run) > len(best_run):
                    best_run = run

        if best_run and len(best_run) >= MIN_RUN_LEN:
            run_count += 1
            all_runs.append(best_run)

            # ===== DIAG: 每个找到的 run =====
            seqs = [r[4] for r in best_run]
            prices = [r[0] for r in best_run]
            diag(f"  找到 run: base=0x{base:X}, 长度={len(best_run)}, "
                 f"seq范围={min(seqs)}-{max(seqs)}, "
                 f"价格范围={min(prices):.2f}-{max(prices):.2f}")
            # ===== DIAG END =====

    # ===== DIAG: full_scan 汇总 =====
    diag_flush()
    diag("DIAG: full_scan 汇总")
    diag(f"  跳过大区域: {skip_count}")
    diag(f"  实际读取区域: {region_count}")
    diag(f"  有候选记录的区域: {candidate_count}")
    diag(f"  找到的连续序列: {run_count}")
    diag(f"  全局共找到 {len(all_runs)} 个连续序列:")
    for i, run in enumerate(all_runs):
        seqs = [r[4] for r in run]
        prices = [r[0] for r in run]
        diag(f"  run[{i}]: 长度={len(run)}, "
             f"seq范围={min(seqs)}-{max(seqs)}, "
             f"价格范围={min(prices):.2f}-{max(prices):.2f}")
    # ===== DIAG END =====

    if not all_runs:
        return None

    result = build_order_book(all_runs)
    if not result:
        return None

    asks, bids, cache_blocks = result
    return asks, bids, cache_blocks


def cached_read(pm, cache_blocks):
    all_runs = []
    for addr, count in cache_blocks:
        try:
            raw = pm.read_bytes(addr, count * RECORD_SIZE)
        except Exception:
            return None
        records = []
        for i in range(count):
            off = i * RECORD_SIZE
            price = struct.unpack_from("<f", raw, off)[0]
            vol, cnt, status, seq = struct.unpack_from("<4I", raw, off + 4)
            if (PRICE_MIN <= price <= PRICE_MAX
                    and 0 < vol < VOL_MAX
                    and cnt < CNT_MAX
                    and 0 <= seq <= MAX_SEQ):
                records.append((price, vol, cnt, status, seq, addr + off))
        runs = find_consecutive_runs(records, min_len=5)
        all_runs.extend(runs)

    if not all_runs:
        return None

    result = build_order_book(all_runs)
    if not result:
        return None

    return result[0], result[1]

"""
千档盘口内存抓取工具 - v10 诊断精简版
只在关键节点写日志，不在循环内写
"""
import pymem
import struct
import time
import ctypes
import ctypes.wintypes
import sys
import os
from collections import Counter
from datetime import datetime

try:
    import numpy as np
    HAS_NUMPY = True
except ImportError:
    HAS_NUMPY = False

PROCESS_NAME = "AX.exe"
RECORD_SIZE = 20
PRICE_MIN = 50.0
PRICE_MAX = 200.0
VOL_MAX = 10_000_000
CNT_MAX = 1_000_000
MAX_SEQ = 2000
MAX_REGION_SIZE = 80 * 1024 * 1024
MIN_RUN_LEN = 10
MEM_COMMIT = 0x1000
PAGE_NOACCESS = 0x01
PAGE_GUARD = 0x100

DIAG_LOG = r"C:\Users\john\ax_diag_output.log"


def diag(msg):
    """仅写文件，不打印（避免拖慢）"""
    try:
        with open(DIAG_LOG, "a", encoding="utf-8") as f:
            f.write(msg + "\n")
    except Exception:
        pass


class MEMORY_BASIC_INFORMATION(ctypes.Structure):
    _fields_ = [
        ("BaseAddress", ctypes.c_void_p),
        ("AllocationBase", ctypes.c_void_p),
        ("AllocationProtect", ctypes.wintypes.DWORD),
        ("RegionSize", ctypes.c_size_t),
        ("State", ctypes.wintypes.DWORD),
        ("Protect", ctypes.wintypes.DWORD),
        ("Type", ctypes.wintypes.DWORD),
    ]


def si(val):
    return int(val) if val is not None else 0


def get_readable_regions(pm):
    regions = []
    addr = 0
    mbi = MEMORY_BASIC_INFORMATION()
    mbi_size = ctypes.sizeof(mbi)
    kernel32 = ctypes.windll.kernel32
    while True:
        ret = kernel32.VirtualQueryEx(
            pm.process_handle, ctypes.c_void_p(addr),
            ctypes.byref(mbi), mbi_size
        )
        if not ret:
            break
        base = si(mbi.BaseAddress)
        size = si(mbi.RegionSize)
        state = si(mbi.State)
        protect = si(mbi.Protect)
        if (state == MEM_COMMIT and size > 0
                and protect != PAGE_NOACCESS
                and not (protect & PAGE_GUARD)):
            regions.append((base, size))
        nxt = base + size
        if nxt <= addr or nxt >= 0x7FFFFFFFFFFF:
            break
        addr = nxt
    return regions


def numpy_find_candidates(raw_bytes):
    if len(raw_bytes) < RECORD_SIZE * 3:
        return []
    buf = bytes(raw_bytes)
    if len(buf) % 4 != 0:
        buf = buf[: len(buf) & ~3]
    all_u32 = np.frombuffer(buf, dtype="<u4")
    all_f32 = np.frombuffer(buf, dtype="<f4")
    n = len(all_u32) - 14
    if n <= 0:
        return []
    p0 = all_f32[0:n]
    v0 = all_u32[1:n + 1]
    s0 = all_u32[4:n + 4]
    p1 = all_f32[5:n + 5]
    v1 = all_u32[6:n + 6]
    s1 = all_u32[9:n + 9]
    p2 = all_f32[10:n + 10]
    v2 = all_u32[11:n + 11]
    s2 = all_u32[14:n + 14]
    mask = (
        (p0 >= PRICE_MIN) & (p0 <= PRICE_MAX) & (v0 > 0) & (v0 < VOL_MAX) & (s0 < MAX_SEQ)
        & (p1 >= PRICE_MIN) & (p1 <= PRICE_MAX) & (v1 > 0) & (v1 < VOL_MAX) & (s1 < MAX_SEQ)
        & (p2 >= PRICE_MIN) & (p2 <= PRICE_MAX) & (v2 > 0) & (v2 < VOL_MAX) & (s2 < MAX_SEQ)
        & (s1 == s0 + 1)
        & (s2 == s0 + 2)
    )
    indices = np.where(mask)[0]
    results = []
    for idx in indices[:50]:
        i = int(idx)
        results.append((i * 4, int(s0[i])))
    return results


def fallback_find_candidates(raw_bytes):
    results = []
    limit = len(raw_bytes) - RECORD_SIZE * 3
    for off in range(0, limit, 4):
        p0 = struct.unpack_from("<f", raw_bytes, off)[0]
        v0, _, _, s0 = struct.unpack_from("<4I", raw_bytes, off + 4)
        if not (PRICE_MIN <= p0 <= PRICE_MAX and 0 < v0 < VOL_MAX and s0 < MAX_SEQ):
            continue
        p1 = struct.unpack_from("<f", raw_bytes, off + RECORD_SIZE)[0]
        v1, _, _, s1 = struct.unpack_from("<4I", raw_bytes, off + RECORD_SIZE + 4)
        if not (PRICE_MIN <= p1 <= PRICE_MAX and 0 < v1 < VOL_MAX and s1 == s0 + 1):
            continue
        p2 = struct.unpack_from("<f", raw_bytes, off + RECORD_SIZE * 2)[0]
        v2, _, _, s2 = struct.unpack_from("<4I", raw_bytes, off + RECORD_SIZE * 2 + 4)
        if not (PRICE_MIN <= p2 <= PRICE_MAX and 0 < v2 < VOL_MAX and s2 == s0 + 2):
            continue
        results.append((off, s0))
        if len(results) >= 50:
            break
    return results


def extract_records(raw_bytes, base_addr, alignment):
    records = []
    for i in range(alignment, len(raw_bytes) - RECORD_SIZE + 1, RECORD_SIZE):
        price = struct.unpack_from("<f", raw_bytes, i)[0]
        vol, cnt, status, seq = struct.unpack_from("<4I", raw_bytes, i + 4)
        if (PRICE_MIN <= price <= PRICE_MAX
                and 0 < vol < VOL_MAX
                and cnt < CNT_MAX
                and 0 <= seq <= MAX_SEQ):
            records.append((price, vol, cnt, status, seq, base_addr + i))
    return records


def find_consecutive_runs(records, min_len=MIN_RUN_LEN):
    if len(records) < min_len:
        return []
    runs = []
    cur = [records[0]]
    for i in range(1, len(records)):
        if records[i][4] == cur[-1][4] + 1:
            cur.append(records[i])
        else:
            if len(cur) >= min_len:
                runs.append(cur)
            cur = [records[i]]
    if len(cur) >= min_len:
        runs.append(cur)
    return runs


def validate_run(run):
    if len(run) < 3:
        return False
    prices = [r[0] for r in run]
    increasing = all(prices[i] <= prices[i + 1] for i in range(len(prices) - 1))
    decreasing = all(prices[i] >= prices[i + 1] for i in range(len(prices) - 1))
    return increasing or decreasing


def deduplicate(records):
    seen = set()
    unique = []
    for r in records:
        key = (round(r[0], 4), r[1], r[4])
        if key not in seen:
            seen.add(key)
            unique.append(r)
    return unique


def split_multi_stock_runs(run):
    if len(run) < 30:
        return [run]
    prices = [r[0] for r in run]
    n = len(prices)
    gaps = sorted(abs(prices[i] - prices[i - 1]) for i in range(1, n))
    median_gap = gaps[len(gaps) // 2] if gaps else 0.10
    threshold = max(median_gap * 10, 2.0)
    split_points = [0]
    for i in range(1, n):
        if prices[i] > prices[i - 1] + threshold:
            split_points.append(i)
    split_points.append(n)
    sub_runs = []
    for i in range(len(split_points) - 1):
        start = split_points[i]
        end = split_points[i + 1]
        if end - start >= 20:
            sub_runs.append(run[start:end])
    return sub_runs if sub_runs else [run]


def find_best_split(run):
    n = len(run)
    if n < 20:
        return None

    prices = [r[0] for r in run]

    prefix_min = [0.0] * n
    suffix_max = [0.0] * n
    prefix_min[0] = prices[0]
    for i in range(1, n):
        prefix_min[i] = min(prefix_min[i - 1], prices[i])
    suffix_max[n - 1] = prices[n - 1]
    for i in range(n - 2, -1, -1):
        suffix_max[i] = max(suffix_max[i + 1], prices[i])

    best_idx = None
    best_balance = -1

    for i in range(10, n - 10):
        ask_min = prefix_min[i - 1]
        bid_max = suffix_max[i]
        if ask_min <= bid_max:
            continue
        spread = ask_min - bid_max
        if spread > 5.0:
            continue
        balance = min(i, n - i) / max(i, n - i)
        if balance > best_balance:
            best_balance = balance
            best_idx = i

    return best_idx


def build_order_book(all_runs):
    candidates = []

    all_sub_runs = []
    for run in all_runs:
        all_sub_runs.extend(split_multi_stock_runs(run))

    # ===== DIAG =====
    diag("=" * 60)
    diag(f"DIAG: split结果 输入={len(all_runs)} 输出={len(all_sub_runs)}")
    for i, run in enumerate(all_sub_runs):
        seqs = [r[4] for r in run]
        prices = [r[0] for r in run]
        diag(f"  sub[{i}]: len={len(run)} seq={min(seqs)}-{max(seqs)} "
             f"price={min(prices):.2f}-{max(prices):.2f}")
        diag(f"    front3: {[(round(r[0],2),r[1],r[4]) for r in run[:3]]}")
        diag(f"    back3:  {[(round(r[0],2),r[1],r[4]) for r in run[-3:]]}")
    # ===== DIAG END =====

    # 策略1
    for run_idx, run in enumerate(all_sub_runs):
        if len(run) < 30:
            continue
        prices = [r[0] for r in run]
        n = len(prices)
        down = sum(1 for i in range(n - 1) if prices[i + 1] < prices[i] - 0.001)
        if down < n * 0.3:
            continue

        split = find_best_split(run)

        # ===== DIAG =====
        diag(f"DIAG: 策略1 sub[{run_idx}] len={len(run)} "
             f"down_ratio={down/n:.2f} split={split}")
        if split:
            ask_prices = [r[0] for r in run[:split]]
            bid_prices = [r[0] for r in run[split:]]
            diag(f"  -> 卖方={split}档 {min(ask_prices):.2f}~{max(ask_prices):.2f}")
            diag(f"  -> 买方={len(run)-split}档 {min(bid_prices):.2f}~{max(bid_prices):.2f}")
        # ===== DIAG END =====

        if split and 10 <= split <= len(run) - 10:
            asks = sorted(run[:split], key=lambda r: r[0])
            bids = sorted(run[split:], key=lambda r: -r[0])
            if asks and bids and asks[0][0] > bids[0][0]:
                spread = asks[0][0] - bids[0][0]
                if 0 < spread < 5.0:
                    balance = min(len(asks), len(bids)) / max(1, max(len(asks), len(bids)))
                    if balance >= 0.1:
                        cache = [(run[0][5], len(run))]
                        candidates.append((asks, bids, balance, cache))
                        diag(f"  -> ACCEPTED balance={balance:.4f}")
                    else:
                        diag(f"  -> REJECT balance={balance:.4f}")
                else:
                    diag(f"  -> REJECT spread={spread:.2f}")
            else:
                diag(f"  -> REJECT 卖一<=买一 or empty")

    # 策略2
    asc_runs = []
    desc_runs = []
    for run in all_sub_runs:
        if len(run) < 10:
            continue
        prices = [r[0] for r in run]
        n = len(prices)
        up = sum(1 for i in range(n - 1) if prices[i + 1] > prices[i] + 0.001)
        down = sum(1 for i in range(n - 1) if prices[i + 1] < prices[i] - 0.001)
        if up > down:
            asc_runs.append(run)
        else:
            desc_runs.append(run)

    # ===== DIAG =====
    diag(f"DIAG: 策略2 asc={len(asc_runs)} desc={len(desc_runs)}")
    for i, run in enumerate(asc_runs):
        prices = [r[0] for r in run]
        diag(f"  asc[{i}]: len={len(run)} price={min(prices):.2f}~{max(prices):.2f}")
    for i, run in enumerate(desc_runs):
        prices = [r[0] for r in run]
        diag(f"  desc[{i}]: len={len(run)} price={min(prices):.2f}~{max(prices):.2f}")
    # ===== DIAG END =====

    for a_run in asc_runs:
        a_min = min(r[0] for r in a_run)
        for b_run in desc_runs:
            b_max = max(r[0] for r in b_run)
            if a_min > b_max:
                spread = a_min - b_max
                if 0 < spread < 5.0:
                    asks = sorted(a_run, key=lambda r: r[0])
                    bids = sorted(b_run, key=lambda r: -r[0])
                    balance = min(len(asks), len(bids)) / max(1, max(len(asks), len(bids)))
                    if balance >= 0.1:
                        cache = [
                            (a_run[0][5], len(a_run)),
                            (b_run[0][5], len(b_run)),
                        ]
                        candidates.append((asks, bids, balance, cache))
                        diag(f"  策略2 ACCEPTED: 卖={len(asks)} 买={len(bids)} "
                             f"bal={balance:.4f}")

    # ===== DIAG =====
    diag(f"DIAG: 最终候选={len(candidates)}")
    for i, c in enumerate(candidates):
        diag(f"  [{i}] 卖={len(c[0])} 买={len(c[1])} bal={c[2]:.4f} "
             f"卖价={min(r[0] for r in c[0]):.2f}~{max(r[0] for r in c[0]):.2f} "
             f"买价={min(r[0] for r in c[1]):.2f}~{max(r[0] for r in c[1]):.2f}")
    # ===== DIAG END =====

    if not candidates:
        return None

    best = max(candidates, key=lambda c: c[2])

    # ===== DIAG: 最终选择 =====
    diag(f"DIAG: 最终选择 卖={len(best[0])} 买={len(best[1])}")

    diag(f"  买方价格: {min(r[0] for r in best[1]):.2f} ~ {max(r[0] for r in best[1]):.2f}")
    # 卖方高价端5档
    asks_hi = sorted(best[0], key=lambda r: -r[0])[:5]
    diag(f"  卖方最高5档:")
    for r in asks_hi:
        diag(f"    price={r[0]:.2f} vol={r[1]} cnt={r[2]} seq={r[4]}")
    # 卖方低价端5档
    asks_lo = sorted(best[0], key=lambda r: r[0])[:5]
    diag(f"  卖方最低5档:")
    for r in asks_lo:
        diag(f"    price={r[0]:.2f} vol={r[1]} cnt={r[2]} seq={r[4]}")

    # 买方低价端5档

def full_scan(pm, regions):
    all_runs = []
    total = len(regions)

    diag("=" * 60)
    diag(f"DIAG: full_scan 开始, 区域数={total}")

    region_count = 0
    skip_count = 0
    candidate_count = 0
    run_count = 0

    for idx, (base, size) in enumerate(regions):
        if size > MAX_REGION_SIZE:
            skip_count += 1
            continue
        if idx % 300 == 0 and idx > 0:
            print(f"  扫描进度: {idx}/{total} ...")
        try:
            raw = pm.read_bytes(base, size)
        except Exception:
            continue

        region_count += 1
        candidates = (numpy_find_candidates(raw) if HAS_NUMPY
                      else fallback_find_candidates(raw))
        if not candidates:
            continue

        candidate_count += 1
        align_counts = Counter(off % RECORD_SIZE for off, _ in candidates)
        best_align = align_counts.most_common(1)[0][0]
        records = extract_records(raw, base, best_align)
        runs = find_consecutive_runs(records)

        best_run = None
        for run in runs:
            if validate_run(run):
                if best_run is None or len(run) > len(best_run):
                    best_run = run

        if best_run and len(best_run) >= MIN_RUN_LEN:
            run_count += 1
            all_runs.append(best_run)
            seqs = [r[4] for r in best_run]
            prices = [r[0] for r in best_run]
            diag(f"  run: base=0x{base:X} len={len(best_run)} "
                 f"seq={min(seqs)}-{max(seqs)} "
                 f"price={min(prices):.2f}-{max(prices):.2f}")

    # ===== DIAG =====
    diag(f"DIAG: scan汇总 skip={skip_count} read={region_count} "
         f"candidate_regions={candidate_count} runs={run_count}")
    diag(f"  共{len(all_runs)}个序列:")
    for i, run in enumerate(all_runs):
        seqs = [r[4] for r in run]
        prices = [r[0] for r in run]
        diag(f"  run[{i}]: len={len(run)} seq={min(seqs)}-{max(seqs)} "
             f"price={min(prices):.2f}-{max(prices):.2f}")
    # ===== DIAG END =====

    if not all_runs:
        return None

    result = build_order_book(all_runs)
    if not result:
        return None

    asks, bids, cache_blocks = result
    return asks, bids, cache_blocks


def cached_read(pm, cache_blocks):
    all_runs = []
    for addr, count in cache_blocks:
        try:
            raw = pm.read_bytes(addr, count * RECORD_SIZE)
        except Exception:
            return None
        records = []
        for i in range(count):
            off = i * RECORD_SIZE
            price = struct.unpack_from("<f", raw, off)[0]
            vol, cnt, status, seq = struct.unpack_from("<4I", raw, off + 4)
            if (PRICE_MIN <= price <= PRICE_MAX
                    and 0 < vol < VOL_MAX
                    and cnt < CNT_MAX
                    and 0 <= seq <= MAX_SEQ):
                records.append((price, vol, cnt, status, seq, addr + off))
        runs = find_consecutive_runs(records, min_len=5)
        all_runs.extend(runs)

    if not all_runs:
        return None

    result = build_order_book(all_runs)
    if not result:
        return None

    return result[0], result[1]
