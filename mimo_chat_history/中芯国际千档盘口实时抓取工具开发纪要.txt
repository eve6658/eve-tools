import pymem
import struct
import subprocess
import csv
import time
import os
import sys
import bisect

# ══════════════════════════════════════════════════════════
#  配置
# ══════════════════════════════════════════════════════════
RECORD_SIZE   = 20       # 5×uint32 = 20字节/条
MIN_SEQ_LEN   = 10       # 最短有效序列
MAX_LEN       = 600      # ★ 600档，覆盖你的545卖+422买
MAX_BACKWARD  = 3        # 卖方回退检查
MAX_CROSS     = 3        # 最多跨3个内存区域
DISPLAY_ROWS  = 80       # 终端显示前80档
SCAN_INTERVAL = 25       # 扫描间隔（秒）


# ══════════════════════════════════════════════════════════
#  进程定位
# ══════════════════════════════════════════════════════════
def find_ax_pid():
    result = subprocess.run(
        ["tasklist", "/FI", "IMAGENAME eq AX.exe", "/FO", "CSV"],
        capture_output=True, text=True
    )
    for line in csv.reader(result.stdout.strip().split("\n")):
        if len(line) >= 2 and line[0].strip('"') == "AX.exe":
            return int(line[1].strip('"'))
    return None


# ══════════════════════════════════════════════════════════
#  内存区域（含跨区域读取）
# ══════════════════════════════════════════════════════════
def enum_readable_regions(pm):
    regions = []
    addr = 0
    min_size = RECORD_SIZE * MIN_SEQ_LEN  # 200字节以下跳过
    while addr < 0x7FFFFFFFFFFF:
        try:
            info = pm.query(addr)
            base = info.BaseAddress
            size = info.RegionSize
            protect = info.Protect
            if (info.State == 0x1000 and
                    protect in (0x02, 0x04, 0x06, 0x20, 0x40, 0x60, 0x80)):
                if size >= min_size:
                    regions.append((base, size))
            addr = base + size
        except:
            addr += 0x1000
    return regions


def build_ridx(regions):
    """构建排序索引，用于跨区域二分查找"""
    sr = sorted(regions, key=lambda r: r[0])
    bases = [r[0] for r in sr]
    return sr, bases


def find_region(ridx, addr):
    sr, bases = ridx
    i = bisect.bisect_right(bases, addr) - 1
    if i < 0:
        return None
    b, s = sr[i]
    return (b, s) if b <= addr < b + s else None


def read_cross(pm, ridx, addr, length):
    """跨内存区域读取——600档=12KB，很可能跨越区域边界"""
    result = b""
    remaining = length
    cur = addr
    for _ in range(MAX_CROSS):
        reg = find_region(ridx, cur)
        if reg is None:
            break
        b, s = reg
        avail = s - (cur - b)
        n = min(remaining, avail)
        try:
            result += pm.read_bytes(cur, n)
        except:
            break
        remaining -= n
        if remaining <= 0:
            break
        cur = b + s
    return result


# ══════════════════════════════════════════════════════════
#  数据解析
# ══════════════════════════════════════════════════════════
def parse_rec(data):
    v = struct.unpack("<5I", data)
    p = struct.unpack("<f", struct.pack("<I", v[0]))[0]
    return p, v[1], v[2], v[3], v[4]   # price, vol, cnt, status, seq


def ok_price(p):
    return 0.01 < p < 100000.0


def ok_step(p1, p2):
    d = abs(p1 - p2)
    return 0.001 < d < 1.0


# ══════════════════════════════════════════════════════════
#  序列构建
# ══════════════════════════════════════════════════════════
def build_seq(chunk, base, offset, first, ridx, pm, ascending):
    """
    从 offset 开始构建连续序列。
    ascending=True  → 卖方（价格递增）
    ascending=False → 买方（价格递减）
    """
    recs = [first]
    prev = first[0]
    seq0 = first[4]

    for j in range(1, MAX_LEN):
        # 优先从当前chunk读，不够则跨区域
        local = offset + j * RECORD_SIZE
        if local + RECORD_SIZE <= len(chunk):
            try:
                rec = parse_rec(chunk[local:local + RECORD_SIZE])
            except:
                break
        else:
            data = read_cross(pm, ridx, base + offset + j * RECORD_SIZE,
                              RECORD_SIZE)
            if len(data) < RECORD_SIZE:
                break
            try:
                rec = parse_rec(data)
            except:
                break

        rp, rv, rc, rs, rseq = rec

        # ★ seq 必须匹配当前快照（±1容差）
        if abs(rseq - seq0) > 1:
            break
        if not ok_price(rp):
            break
        # 价格单调性
        if ascending:
            if rp <= prev:
                break
        else:
            if rp >= prev:
                break
        if not ok_step(rp, prev):
            break

        recs.append(rec)
        prev = rp

    return recs


def expand_ask_backward(chunk, offset, ask_recs):
    """卖方回退：检查前1~3条记录是否属于同一序列（修复第一档缺失）"""
    if not ask_recs:
        return ask_recs

    fp = ask_recs[0][0]
    fs = ask_recs[0][4]
    step = ask_recs[1][0] - ask_recs[0][0] if len(ask_recs) >= 2 else 0.01

    expanded = list(ask_recs)

    for back in range(1, MAX_BACKWARD + 1):
        co = offset - back * RECORD_SIZE
        if co < 0:
            break
        try:
            rec = parse_rec(chunk[co:co + RECORD_SIZE])
        except:
            break

        p, v, c, s, sq = rec
        if not ok_price(p) or v == 0:
            break
        if abs(sq - fs) > 1:
            break
        if p >= fp:
            break
        d = fp - p
        if not (0.001 < d < 1.0):
            break
        if step > 0 and back > 1:
            ratio = d / step
            if ratio < 0.5 or ratio > 1.5:
                break

        expanded.insert(0, rec)
        fp = p

    return expanded


# ══════════════════════════════════════════════════════════
#  主搜索
# ══════════════════════════════════════════════════════════
def search_books(pm, regions, ridx):
    all_asks = []
    all_bids = []

    for base, size in regions:
        try:
            chunk = pm.read_bytes(base, size)
        except:
            continue

        i = 0
        while i <= size - RECORD_SIZE:
            try:
                first = parse_rec(chunk[i:i + RECORD_SIZE])
            except:
                i += 4
                continue

            p0, v0 = first[0], first[1]
            if not ok_price(p0) or v0 == 0:
                i += RECORD_SIZE
                continue

            # ── 卖方（递增）──
            asks = build_seq(chunk, base, i, first, ridx, pm, ascending=True)
            if len(asks) >= MIN_SEQ_LEN:
                expanded = expand_ask_backward(chunk, i, asks)
                all_asks.append((first[4], base + i, expanded))
                i += len(asks) * RECORD_SIZE   # 跳过已匹配区域
                continue

            # ── 买方（递减）──
            bids = build_seq(chunk, base, i, first, ridx, pm, ascending=False)
            if len(bids) >= MIN_SEQ_LEN:
                all_bids.append((first[4], base + i, bids))
                i += len(bids) * RECORD_SIZE
                continue

            i += RECORD_SIZE

    return all_asks, all_bids


# ══════════════════════════════════════════════════════════
#  深层大单检测
# ══════════════════════════════════════════════════════════
def find_notable(records, label, threshold=10):
    """找出深层档位中挂单量 > 平均值 threshold 倍的价位"""
    if len(records) <= DISPLAY_ROWS:
        return []
    deep = records[DISPLAY_ROWS:]
    vols = [r[1] for r in deep if r[1] > 0]
    if not vols:
        return []
    avg = sum(vols) / len(vols)
    if avg == 0:
        return []
    result = []
    for i, r in enumerate(deep):
        if r[1] > avg * threshold:
            level = DISPLAY_ROWS + i + 1
            result.append((label, level, r[0], r[1], r[1] / avg))
    return sorted(result, key=lambda x: -x[3])

    print("-" * 80)

    rows = min(DISPLAY_ROWS, max(len(bids), len(asks)))
    for i in range(rows):
        if i < len(bids):
            bp, bv, bc, _, _ = bids[i]
            bs = f"  {i+1:>3} {bp:>10.2f} {bv:>10,} {bc:>6}"

    rem_b = len(bids) - rows
    rem_a = len(asks) - rows
    if rem_b > 0 or rem_a > 0:

import pymem
import struct
import subprocess
import csv
import time
import os
import sys
import bisect

# ══════════════════════════════════════════════════════════
#  配置
# ══════════════════════════════════════════════════════════
RECORD_SIZE   = 20
PROBE_LEN     = 12       # ★ 第一阶段：快速探测12条
MAX_LEN       = 600      # 第二阶段：通过探测后扩展到600条
MIN_SEQ_LEN   = 10
MAX_BACKWARD  = 3
MAX_CROSS     = 3
DISPLAY_ROWS  = 80
SCAN_INTERVAL = 25


# ══════════════════════════════════════════════════════════
#  进程
# ══════════════════════════════════════════════════════════
def find_ax_pid():
    result = subprocess.run(
        ["tasklist", "/FI", "IMAGENAME eq AX.exe", "/FO", "CSV"],
        capture_output=True, text=True
    )
    for line in csv.reader(result.stdout.strip().split("\n")):
        if len(line) >= 2 and line[0].strip('"') == "AX.exe":
            return int(line[1].strip('"'))
    return None


# ══════════════════════════════════════════════════════════
#  内存
# ══════════════════════════════════════════════════════════
def enum_readable_regions(pm):
    regions = []
    addr = 0
    min_size = RECORD_SIZE * MIN_SEQ_LEN
    while addr < 0x7FFFFFFFFFFF:
        try:
            info = pm.query(addr)
            base = info.BaseAddress
            size = info.RegionSize
            protect = info.Protect
            if (info.State == 0x1000 and
                    protect in (0x02, 0x04, 0x06, 0x20, 0x40, 0x60, 0x80)):
                if size >= min_size:
                    regions.append((base, size))
            addr = base + size
        except:
            addr += 0x1000
    return regions


def build_ridx(regions):
    sr = sorted(regions, key=lambda r: r[0])
    return sr, [r[0] for r in sr]


def find_region(ridx, addr):
    sr, bases = ridx
    i = bisect.bisect_right(bases, addr) - 1
    if i < 0:
        return None
    b, s = sr[i]
    return (b, s) if b <= addr < b + s else None


def read_cross(pm, ridx, addr, length):
    result = b""
    remaining = length
    cur = addr
    for _ in range(MAX_CROSS):
        reg = find_region(ridx, cur)
        if reg is None:
            break
        b, s = reg
        n = min(remaining, s - (cur - b))
        try:
            result += pm.read_bytes(cur, n)
        except:
            break
        remaining -= n
        if remaining <= 0:
            break
        cur = b + s
    return result


# ══════════════════════════════════════════════════════════
#  解析
# ══════════════════════════════════════════════════════════
def parse_rec(data):
    v = struct.unpack("<5I", data)
    p = struct.unpack("<f", struct.pack("<I", v[0]))[0]
    return p, v[1], v[2], v[3], v[4]


def ok_price(p):
    return 0.01 < p < 100000.0


def ok_step(p1, p2):
    d = abs(p1 - p2)
    return 0.001 < d < 1.0


# ══════════════════════════════════════════════════════════
#  ★ 两阶段序列构建
# ══════════════════════════════════════════════════════════
def probe_seq(chunk, offset, first, ascending):
    """
    第一阶段：快速探测 PROBE_LEN 条。
    全部在当前chunk内完成，不跨区域，极快。
    返回 (passed, count, last_price, seq0)
    """
    prev = first[0]
    seq0 = first[4]
    count = 1

    for j in range(1, PROBE_LEN):
        local = offset + j * RECORD_SIZE
        if local + RECORD_SIZE > len(chunk):
            return count >= MIN_SEQ_LEN, count, prev, seq0
        try:
            rec = parse_rec(chunk[local:local + RECORD_SIZE])
        except:
            return False, count, prev, seq0

        rp, rv, rc, rs, rseq = rec
        if abs(rseq - seq0) > 1:
            return count >= MIN_SEQ_LEN, count, prev, seq0
        if not ok_price(rp):
            return False, count, prev, seq0
        if ascending:
            if rp <= prev:
                return count >= MIN_SEQ_LEN, count, prev, seq0
        else:
            if rp >= prev:
                return count >= MIN_SEQ_LEN, count, prev, seq0
        if not ok_step(rp, prev):
            return False, count, prev, seq0

        count += 1
        prev = rp

    # 探测通过，返回 True
    return True, count, prev, seq0


def expand_seq(chunk, base, offset, first, probe_count,
               last_price, seq0, ridx, pm, ascending):
    """
    第二阶段：从 probe_count 处继续扩展到 MAX_LEN。
    可能跨区域。
    """
    recs = []
    # 先把探测过的记录解析出来
    for j in range(probe_count):
        local = offset + j * RECORD_SIZE
        try:
            rec = parse_rec(chunk[local:local + RECORD_SIZE])
            recs.append(rec)
        except:
            break

    prev = last_price

    # 从 probe_count 继续
    for j in range(probe_count, MAX_LEN):
        local = offset + j * RECORD_SIZE
        if local + RECORD_SIZE <= len(chunk):
            try:
                rec = parse_rec(chunk[local:local + RECORD_SIZE])
            except:
                break
        else:
            data = read_cross(pm, ridx, base + offset + j * RECORD_SIZE,
                              RECORD_SIZE)
            if len(data) < RECORD_SIZE:
                break
            try:
                rec = parse_rec(data)
            except:
                break

        rp, rv, rc, rs, rseq = rec
        if abs(rseq - seq0) > 1:
            break
        if not ok_price(rp):
            break
        if ascending:
            if rp <= prev:
                break
        else:
            if rp >= prev:
                break
        if not ok_step(rp, prev):
            break

        recs.append(rec)
        prev = rp

    return recs


def expand_ask_backward(chunk, offset, ask_recs):
    if not ask_recs:
        return ask_recs
    fp = ask_recs[0][0]
    fs = ask_recs[0][4]
    step = ask_recs[1][0] - ask_recs[0][0] if len(ask_recs) >= 2 else 0.01
    expanded = list(ask_recs)
    for back in range(1, MAX_BACKWARD + 1):
        co = offset - back * RECORD_SIZE
        if co < 0:
            break
        try:
            rec = parse_rec(chunk[co:co + RECORD_SIZE])
        except:
            break
        p, v, c, s, sq = rec
        if not ok_price(p) or v == 0:
            break
        if abs(sq - fs) > 1:
            break
        if p >= fp:
            break
        d = fp - p
        if not (0.001 < d < 1.0):
            break
        if step > 0 and back > 1:
            ratio = d / step
            if ratio < 0.5 or ratio > 1.5:
                break
        expanded.insert(0, rec)
        fp = p
    return expanded


# ══════════════════════════════════════════════════════════
#  主搜索
# ══════════════════════════════════════════════════════════
def search_books(pm, regions, ridx):
    all_asks = []
    all_bids = []
    total = len(regions)

    for ri, (base, size) in enumerate(regions):
        # 进度
        if ri % 200 == 0:
            print(f"\r  扫描进度: {ri}/{total} 区域 "
                  f"({len(all_asks)}卖 {len(all_bids)}买)",
                  end="", flush=True)

        try:
            chunk = pm.read_bytes(base, size)
        except:
            continue

        i = 0
        while i <= size - RECORD_SIZE:
            try:
                first = parse_rec(chunk[i:i + RECORD_SIZE])
            except:
                i += 4
                continue

            p0, v0 = first[0], first[1]
            if not ok_price(p0) or v0 == 0:
                i += RECORD_SIZE
                continue

            # ── 第一阶段：快速探测卖方 ──
            passed, cnt, lp, sq = probe_seq(chunk, i, first, ascending=True)
            if passed and cnt >= MIN_SEQ_LEN:
                # 第二阶段：扩展到600
                asks = expand_seq(chunk, base, i, first, cnt,
                                  lp, sq, ridx, pm, ascending=True)
                if len(asks) >= MIN_SEQ_LEN:
                    expanded = expand_ask_backward(chunk, i, asks)
                    all_asks.append((sq, base + i, expanded))
                    i += len(asks) * RECORD_SIZE
                    continue

            # ── 第一阶段：快速探测买方 ──
            passed, cnt, lp, sq = probe_seq(chunk, i, first, ascending=False)
            if passed and cnt >= MIN_SEQ_LEN:
                bids = expand_seq(chunk, base, i, first, cnt,
                                  lp, sq, ridx, pm, ascending=False)
                if len(bids) >= MIN_SEQ_LEN:
                    all_bids.append((sq, base + i, bids))
                    i += len(bids) * RECORD_SIZE
                    continue

            i += RECORD_SIZE

    print(f"\r  扫描完成: {total} 区域, "
          f"{len(all_asks)}个卖方, {len(all_bids)}个买方        ")
    return all_asks, all_bids


# ══════════════════════════════════════════════════════════
#  深层大单
# ══════════════════════════════════════════════════════════
def find_notable(records, label, threshold=10):
    if len(records) <= DISPLAY_ROWS:
        return []
    deep = records[DISPLAY_ROWS:]
    vols = [r[1] for r in deep if r[1] > 0]
    if not vols:
        return []
    avg = sum(vols) / len(vols)
    if avg == 0:
        return []
    result = []
    for i, r in enumerate(deep):
        if r[1] > avg * threshold:
            lvl = DISPLAY_ROWS + i + 1
            result.append((label, lvl, r[0], r[1], r[1] / avg))
    return sorted(result, key=lambda x: -x[3])

def display(bids, asks, scan_time):
    os.system("cls" if os.name == "nt" else "clear")

    print("=" * 80)
    print("  千档盘口实时监控  MAX_LEN=600  两阶段扫描")
    print("=" * 80)

    if not bids and not asks:
        print("\n  等待数据...\n")
        return

    last = asks[0][0] if asks else (bids[0][0] if bids else 0)
    print(f"  最新价: {last:.2f}    买方: {len(bids)}档    卖方: {len(asks)}档")

    if asks:
        print(f"  卖方: {asks[0][0]:.2f} ~ {asks[-1][0]:.2f}  "
              f"(跨度 {asks[-1][0] - asks[0][0]:.2f} HKD)")
    if bids:
        print(f"  买方: {bids[-1][0]:.2f} ~ {bids[0][0]:.2f}  "
              f"(跨度 {bids[0][0] - bids[-1][0]:.2f} HKD)")

    print("-" * 80)
    print(f"  {'买方':^34}  │  {'卖方':^34}")
    print(f"  {'#':>3} {'价格':>10} {'量':>10} {'笔':>6}  │"
          f"  {'#':>3} {'价格':>10} {'量':>10} {'笔':>6}")
    print("-" * 80)

    rows = min(DISPLAY_ROWS, max(len(bids), len(asks)))
    for i in range(rows):
        if i < len(bids):
            bp, bv, bc, _, _ = bids[i]
            bs = f"  {i+1:>3} {bp:>10.2f} {bv:>10,} {bc:>6}"
        else:
            bs = " " * 35
        if i < len(asks):
            ap, av, ac, _, _ = asks[i]
            a_s = f"  {i+1:>3} {ap:>10.2f} {av:>10,} {ac:>6}"
        else:
            a_s = " " * 35
        print(f"{bs}  │ {a_s}")

    rem_b = len(bids) - rows
    rem_a = len(asks) - rows
    if rem_b > 0 or rem_a > 0:
        print(f"\n  ... 买方还 {rem_b} 档 / 卖方还 {rem_a} 档未显示 ...")

    notable = find_notable(bids, "买") + find_notable(asks, "卖")
    if notable:
        print()
        print("  ★ 深层大单:")
        for side, lvl, price, vol, ratio in notable:
            print(f"    {side}方 #{lvl:<4} {price:>10.2f}  "
                  f"量={vol:>10,}  ({ratio:.0f}x平均)")

    print("-" * 80)
    print(f"  {time.strftime('%H:%M:%S')}  扫描耗时 {scan_time:.1f}s  "
          f"显示1s / 扫描{SCAN_INTERVAL}s")

def main():
    print("定位 AX.exe ...")
    pid = find_ax_pid()
    if not pid:
        print("错误: 未找到 AX.exe")
        sys.exit(1)
    print(f"PID: {pid}")

    pm = pymem.Pymem()
    pm.open_process_from_id(pid)

    print("枚举内存区域...")
    regions = enum_readable_regions(pm)
    ridx = build_ridx(regions)
    print(f"共 {len(regions)} 个可读区域")
    print(f"开始扫描...\n")

    last_scan = 0
    scan_time = 0
    best_bids, best_asks = [], []

    while True:
        now = time.time()
        if now - last_scan >= SCAN_INTERVAL:
            t0 = time.time()
            all_asks, all_bids = search_books(pm, regions, ridx)
            scan_time = time.time() - t0
            last_scan = time.time()

            best_asks = (sorted(all_asks, key=lambda x: x[0], reverse=True)
                         [0][2] if all_asks else [])
            best_bids = (sorted(all_bids, key=lambda x: x[0], reverse=True)
                         [0][2] if all_bids else [])

        display(best_bids, best_asks, scan_time)
        time.sleep(1)


if __name__ == "__main__":
    main()

#  配置

RECORD_SIZE   = 20
MAX_LEN       = 600
MIN_SEQ_LEN   = 10
MAX_BACKWARD  = 3
DISPLAY_ROWS  = 80
SCAN_INTERVAL = 30

# 区域过滤
REGION_MIN    = 5000       # 5KB 以下跳过
REGION_MAX    = 100_000_000  # 100MB 以上跳过

# 快速预筛参数
PRICE_MIN     = 0.5
PRICE_MAX     = 10000.0
STEP_MIN      = 0.001
STEP_MAX      = 1.0

# ══════════════════════════════════════════════════════════

def enum_readable_regions(pm):
    regions = []
    addr = 0
    while addr < 0x7FFFFFFFFFFF:
        try:
            info = pm.query(addr)
            base = info.BaseAddress
            size = info.RegionSize
            protect = info.Protect
            if (info.State == 0x1000 and
                    protect in (0x02, 0x04, 0x06, 0x20, 0x40, 0x60, 0x80)):
                if REGION_MIN <= size <= REGION_MAX:
                    regions.append((base, size))
            addr = base + size
        except:
            addr += 0x1000
    return regions


# ══════════════════════════════════════════════════════════
#  解析
# ══════════════════════════════════════════════════════════
def parse_rec(data):
    v = struct.unpack("<5I", data)
    p = struct.unpack("<f", struct.pack("<I", v[0]))[0]
    return p, v[1], v[2], v[3], v[4]


def ok_price(p):
    return PRICE_MIN < p < PRICE_MAX


def ok_step(p1, p2):
    d = abs(p1 - p2)
    return STEP_MIN < d < STEP_MAX


# ══════════════════════════════════════════════════════════
#  ★ 核心优化：快速预筛
# ══════════════════════════════════════════════════════════
def fast_prescan(chunk):
    """
    用纯字节操作快速扫描整个 chunk。
    找出所有「看起来像连续盘口记录」的起始偏移。
    返回候选偏移列表。

    原理：连续 3 条记录的 price 字段都是有效 float，
    且价格递增/递减、步长合理 → 这个位置值得深入检查。
    """
    candidates = []
    length = len(chunk)
    # 需要至少 3 条记录 = 60 字节
    if length < RECORD_SIZE * 3:
        return candidates

    i = 0
    while i <= length - RECORD_SIZE * 3:
        # 读取第1条的 price 和 vol
        p1_raw = struct.unpack_from("<I", chunk, i)[0]
        v1 = struct.unpack_from("<I", chunk, i + 4)[0]

        # 快速检查 vol > 0（整数比较，极快）
        if v1 == 0:
            i += RECORD_SIZE
            continue

        # 快速检查 price 是否是合理 float
        p1 = struct.unpack_from("<f", chunk, i)[0]
        if not (PRICE_MIN < p1 < PRICE_MAX):
            i += RECORD_SIZE
            continue

        # 读取第2条
        p2 = struct.unpack_from("<f", chunk, i + RECORD_SIZE)[0]
        if not (PRICE_MIN < p2 < PRICE_MAX):
            i += RECORD_SIZE
            continue

        # 检查步长
        d = abs(p2 - p1)
        if not (STEP_MIN < d < STEP_MAX):
            i += RECORD_SIZE
            continue

        # 读取第3条验证
        p3 = struct.unpack_from("<f", chunk, i + RECORD_SIZE * 2)[0]
        if not (PRICE_MIN < p3 < PRICE_MAX):
            i += RECORD_SIZE
            continue

        d2 = abs(p3 - p2)
        if not (STEP_MIN < d2 < STEP_MAX):
            i += RECORD_SIZE
            continue

        # 检查方向一致性
        if (p2 > p1 and p3 > p2) or (p2 < p1 and p3 < p2):
            candidates.append(i)

        i += RECORD_SIZE

    return candidates


# ══════════════════════════════════════════════════════════
#  精确序列构建（只对候选位置执行）
# ══════════════════════════════════════════════════════════
def build_seq(chunk, offset, first, ascending):
    """从候选位置构建完整序列，全在 chunk 内完成"""
    recs = [first]
    prev = first[0]
    seq0 = first[4]

    for j in range(1, MAX_LEN):
        pos = offset + j * RECORD_SIZE
        if pos + RECORD_SIZE > len(chunk):
            break
        try:
            rec = parse_rec(chunk[pos:pos + RECORD_SIZE])
        except:
            break

        rp, rv, rc, rs, rseq = rec
        if abs(rseq - seq0) > 1:
            break
        if not ok_price(rp):
            break
        if ascending and rp <= prev:
            break
        if not ascending and rp >= prev:
            break
        if not ok_step(rp, prev):
            break

        recs.append(rec)
        prev = rp

    return recs


def expand_ask_backward(chunk, offset, ask_recs):
    if not ask_recs:
        return ask_recs
    fp = ask_recs[0][0]
    fs = ask_recs[0][4]
    step = ask_recs[1][0] - ask_recs[0][0] if len(ask_recs) >= 2 else 0.01
    expanded = list(ask_recs)
    for back in range(1, MAX_BACKWARD + 1):
        co = offset - back * RECORD_SIZE
        if co < 0:
            break
        try:
            rec = parse_rec(chunk[co:co + RECORD_SIZE])
        except:
            break
        p, v, c, s, sq = rec
        if not ok_price(p) or v == 0:
            break
        if abs(sq - fs) > 1:
            break
        if p >= fp:
            break
        d = fp - p
        if not (0.001 < d < 1.0):
            break
        if step > 0 and back > 1:
            ratio = d / step
            if ratio < 0.5 or ratio > 1.5:
                break
        expanded.insert(0, rec)
        fp = p
    return expanded


# ══════════════════════════════════════════════════════════
#  主搜索（新架构）
# ══════════════════════════════════════════════════════════
def search_books(pm, regions):
    all_asks = []
    all_bids = []
    total = len(regions)
    total_candidates = 0

    for ri, (base, size) in enumerate(regions):
        if ri % 100 == 0:
            print(f"\r  进度: {ri}/{total} 区域, "
                  f"{total_candidates} 候选, "
                  f"{len(all_asks)}卖 {len(all_bids)}买",
                  end="", flush=True)

        try:
            chunk = pm.read_bytes(base, size)
        except:
            continue

        # ★ 快速预筛：几毫秒扫完整个区域
        candidates = fast_prescan(chunk)
        total_candidates += len(candidates)

        if not candidates:
            continue

        # ★ 只对候选位置做精确解析
        for offset in candidates:
            try:
                first = parse_rec(chunk[offset:offset + RECORD_SIZE])
            except:
                continue

            p0, v0 = first[0], first[1]
            if not ok_price(p0) or v0 == 0:
                continue

            # 卖方
            asks = build_seq(chunk, offset, first, ascending=True)
            if len(asks) >= MIN_SEQ_LEN:
                expanded = expand_ask_backward(chunk, offset, asks)
                all_asks.append((first[4], base + offset, expanded))
                continue

            # 买方
            bids = build_seq(chunk, offset, first, ascending=False)
            if len(bids) >= MIN_SEQ_LEN:
                all_bids.append((first[4], base + offset, bids))

    print(f"\r  完成: {total} 区域, {total_candidates} 候选, "
          f"{len(all_asks)} 卖方序列, {len(all_bids)} 买方序列        ")
    return all_asks, all_bids


# ══════════════════════════════════════════════════════════
#  深层大单
# ══════════════════════════════════════════════════════════
def find_notable(records, label, threshold=10):
    if len(records) <= DISPLAY_ROWS:
        return []
    deep = records[DISPLAY_ROWS:]
    vols = [r[1] for r in deep if r[1] > 0]
    if not vols:
        return []
    avg = sum(vols) / len(vols)
    if avg == 0:
        return []
    result = []
    for i, r in enumerate(deep):
        if r[1] > avg * threshold:
            lvl = DISPLAY_ROWS + i + 1
            result.append((label, lvl, r[0], r[1], r[1] / avg))
    return sorted(result, key=lambda x: -x[3])

二、技术方案


工具栈：Python + pymem + ctypes + struct
原理：通过 pymem 打开 AX.exe 进程，读取其内存，扫描符合盘口数据结构的连续记录
运行环境：Windows 10，管理员 CMD




三、数据结构（已验证确认）



每条盘口记录 20 字节（5 个 uint32）：



偏移	字段	说明
+0	price (float32)	委托价格，如 116.58
+4	vol (uint32)	委托量
+8	cnt (uint32)	笔数
+12	status (uint32)	状态/标志
+16	seq (uint32)	快照序号（同一快照内相同或相近）



内存存储特征：

卖方：价格递增存储（116.58 → 116.59 → 116.60...），第一条即卖一
买方：价格递减存储（116.57 → 116.56 → 116.55...），第一条即买一
环形缓冲区：多个快照连续存储，seq 字段区分不同快照

#	问题	解决方案
1	定位 AX.exe 进程	tasklist + csv 解析获取 PID
